61inline void format_uuid_v1(uuid_upnp* uid, uint16_t clock_seq,
62 uuid_time_t timestamp, uuid_node_t node) {
63 /* Construct a version 1 uuid with the information we've gathered
64 * plus a few constants. */
65 uid->time_low = (uint32_t)(timestamp & 0xFFFFFFFF);
66 uid->time_mid = (uint16_t)((timestamp >> 32) & 0xFFFF);
67 uid->time_hi_and_version = (uint16_t)((timestamp >> 48) & 0x0FFF);
68 uid->time_hi_and_version |= (1 << 12);
69 uid->clock_seq_low = (uint8_t)(clock_seq & 0xFF);
70 uid->clock_seq_hi_and_reserved = (uint8_t)((clock_seq & 0x3F00) >> 8);
71 uid->clock_seq_hi_and_reserved |= 0x80;
72 memcpy(&uid->node, &node, sizeof uid->node);
73}

96inline void write_state(uint16_t clockseq, uuid_time_t timestamp,
97 uuid_node_t node) {
98 static uuid_time_t next_save;
99
100 if (!stateInited) {
101 next_save = timestamp;
102 stateInited = 1;
103 };
104 /* always save state to volatile shared state. */
105 st.cs = clockseq;
106 st.ts = timestamp;
107 st.node = node;
108 if (timestamp >= next_save) {
109 /* schedule next save for 10 seconds from now. */
110 next_save = timestamp + (10 * 10 * 1000 * 1000);
111 };
112}

119inline void get_current_time(uuid_time_t* timestamp) {
120 uuid_time_t time_now;
121 static uuid_time_t time_last;
122 static uint16_t uuids_this_tick;
123 static int inited = 0;
124
125 if (!inited) {
126 uuids_this_tick = UUIDS_PER_TICK;
127 inited = 1;
128 };
129 while (1) {
130 get_system_time(&time_now);
131 /* if clock reading changed since last UUID generated... */
132 if (time_last != time_now) {
133 /* reset count of uuids gen'd with this clock reading.
134 */
135 uuids_this_tick = 0;
136 break;
137 };
138 if (uuids_this_tick < UUIDS_PER_TICK) {
139 uuids_this_tick++;
140 break;
141 };
142 /* going too fast for our clock; spin. */
143 };
144 /* add the count of uuids to low order bits of the clock reading. */
145 *timestamp = time_now + uuids_this_tick;
146 time_last = *timestamp;
147}

153inline uint16_t true_random() {
154 static int inited = 0;
155 uuid_time_t time_now;
156
157 if (!inited) {
158 get_system_time(&time_now);
159 time_now = time_now / UUIDS_PER_TICK;
160 srand((unsigned int)(((time_now >> 32) ^ time_now) & 0xffffffff));
161 inited = 1;
162 };
163
164 return (uint16_t)(rand());
165}

170inline void format_uuid_v3(uuid_upnp* uid, unsigned char hash[16]) {
171 /* Construct a version 3 uuid with the (pseudo-)random number plus a few
172 * constants. */
173 memcpy(uid, hash, sizeof(uuid_upnp));
174 /* convert UUID to local byte order. */
175 uid->time_low = ntohl(uid->time_low);
176 uid->time_mid = ntohs(uid->time_mid);
177 uid->time_hi_and_version = ntohs(uid->time_hi_and_version);
178 /* put in the variant and version bits. */
179 uid->time_hi_and_version &= 0x0FFF;
180 uid->time_hi_and_version |= (3 << 12);
181 uid->clock_seq_hi_and_reserved &= 0x3F;
182 uid->clock_seq_hi_and_reserved |= 0x80;
183}

193int uuid_create(uuid_upnp* uid) {
194 uuid_time_t timestamp;
195 uuid_time_t last_time;
196 uint16_t clockseq;
197 uuid_node_t node;
198 uuid_node_t last_node;
199 int f;
200
201 /* acquire system wide lock so we're alone. */
202 UUIDLock();
203 /* get current time. */
204 get_current_time(&timestamp);
205 /* get node ID. */
206 get_ieee_node_identifier(&node);
207 /* get saved state from NV storage. */
208 f = read_state(&clockseq, &last_time, &last_node);
209 /* if no NV state, or if clock went backwards, or node ID changed
210 * (e.g., net card swap) change clockseq. */
211 if (!f || memcmp(&node, &last_node, sizeof(uuid_node_t)))
212 clockseq = true_random();
213 else if (timestamp < last_time)
214 clockseq++;
215 /* stuff fields into the UUID. */
216 format_uuid_v1(uid, clockseq, timestamp, node);
217 /* save the state for next time. */
218 write_state(clockseq, timestamp, node);
219 UUIDUnlock();
220
221 return 1;
222}

232void uuid_create_from_name(
234 uuid_upnp* uid,
237 uuid_upnp nsid,
239 void* name,
241 int namelen) {
242 MD5_CTX c;
243 unsigned char hash[16];
244 uuid_upnp net_nsid; /* context UUID in network byte order */
245
246 /* put name space ID in network byte order so it hashes the same no
247 * matter what endian machine we're on. */
248 net_nsid = nsid;
249 net_nsid.time_low = htonl(net_nsid.time_low);
250 net_nsid.time_mid = htons(net_nsid.time_mid);
251 net_nsid.time_hi_and_version = htons(net_nsid.time_hi_and_version);
252 MD5Init(&c);
253 MD5Update(&c, (unsigned char*)&net_nsid, sizeof(uuid_upnp));
254 MD5Update(&c, name, (unsigned int)namelen);
255 MD5Final(hash, &c);
256 /* the hash is in network byte order at this point. */
257 format_uuid_v3(uid, hash);
258}

260int uuid_compare(uuid_upnp* u1, uuid_upnp* u2) {
261 int i;
262
263 CHECK(u1->time_low, u2->time_low);

267 CHECK(u1->clock_seq_low, u2->clock_seq_low)
268 for (i = 0; i < 6; i++) {
269 if (u1->node[i] < u2->node[i])
270 return -1;
271 if (u1->node[i] > u2->node[i])
272 return 1;
273 }
