44int map_str_to_int(const char* name, size_t name_len,
45 const str_int_entry* table, int num_entries,
46 int case_sensitive) {
47 TRACE("Executing map_str_to_int()");
48 if (name == nullptr || table == nullptr)
49 return -1;
50
51 int top, mid, bot;
52 int cmp;
53 memptr name_ptr;
54
55 name_ptr.buf = (char*)name;
56 name_ptr.length = name_len;
57
58 top = 0;
59 bot = num_entries - 1;
60
61 while (top <= bot) {
62 mid = (top + bot) / 2;
63 if (case_sensitive) {
64 /*cmp = strcmp( name, table[mid].name ); */
65 cmp = memptr_cmp(&name_ptr, table[mid].name);
66 } else {
67 /*cmp = strcasecmp( name, table[mid].name ); */
68 cmp = memptr_cmp_nocase(&name_ptr, table[mid].name);
69 }
70
71 if (cmp > 0) {
72 top = mid + 1; /* look below mid */
73 } else if (cmp < 0) {
74 bot = mid - 1; /* look above mid */
75 } else { /* cmp == 0 */
76 return mid; /* match; return table index */
77 }
78 }
79
80 return -1; /* header name not found */
81}
