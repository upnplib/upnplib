162parse_status_t scanner_get_token( //
163 scanner_t* scanner,
164 memptr* token,
165 token_type_t* tok_type
166) {
167 char* cursor;
168 char* null_terminator; /* point to null-terminator in buffer */
169 int c;
170 token_type_t token_type;
171 int got_end_quote;
172
173 assert(scanner);
174 assert(token);
175 assert(tok_type);
176
177 /* point to next char in buffer */
178 cursor = scanner->msg->buf + scanner->cursor;
179 null_terminator = scanner->msg->buf + scanner->msg->length;
180 /* not enough chars in input to parse */
181 if (cursor == null_terminator)
182 return PARSE_INCOMPLETE;
183 c = *cursor;
184 if (is_identifier_char(c)) {
185 /* scan identifier */
186 token->buf = cursor++;
187 token_type = TT_IDENTIFIER;
188 while (cursor < null_terminator && is_identifier_char(*cursor))
189 cursor++;
190 if (!scanner->entire_msg_loaded && cursor == null_terminator)
191 /* possibly more valid chars */
192 return PARSE_INCOMPLETE;
193 /* calc token length */
194 token->length = (size_t)cursor - (size_t)token->buf;
195 } else if (c == ' ' || c == '\t') {
196 token->buf = cursor++;
197 token_type = TT_WHITESPACE;
198 while (cursor < null_terminator && (*cursor == ' ' || *cursor == '\t'))
199 cursor++;
200 if (!scanner->entire_msg_loaded && cursor == null_terminator)
201 /* possibly more chars */
202 return PARSE_INCOMPLETE;
203 token->length = (size_t)cursor - (size_t)token->buf;
204 } else if (c == TOKCHAR_CR) {
205 /* scan CRLF */
206 token->buf = cursor++;
207 if (cursor == null_terminator)
208 /* not enuf info to determine CRLF */
209 return PARSE_INCOMPLETE;
210 if (*cursor != TOKCHAR_LF) {
211 /* couldn't match CRLF; match as CR */
212 token_type = TT_CTRL; /* ctrl char */
213 token->length = (size_t)1;
214 } else {
215 /* got CRLF */
216 token->length = (size_t)2;
217 token_type = TT_CRLF;
218 cursor++;
219 }
220 } else if (c == TOKCHAR_LF) { /* accept \n as CRLF */
221 token->buf = cursor++;
222 token->length = (size_t)1;
223 token_type = TT_CRLF;
224 } else if (c == '"') {
225 /* quoted text */
226 token->buf = cursor++;
227 token_type = TT_QUOTEDSTRING;
228 got_end_quote = 0;
229 while (cursor < null_terminator) {
230 c = *cursor++;
231 if (c == '"') {
232 got_end_quote = 1;
233 break;
234 } else if (c == '\\') {
235 if (cursor < null_terminator) {
236 c = *cursor++;
237 /* the char after '\\' could be ANY
238 * octet */
239 }
240 /* else, while loop handles incomplete buf */
241 } else if (is_qdtext_char(c)) {
242 /* just accept char */
243 } else
244 /* bad quoted text */
245 return PARSE_FAILURE;
246 }
247 if (got_end_quote)
248 token->length = (size_t)cursor - (size_t)token->buf;
249 else { /* incomplete */
250
251 assert(cursor == null_terminator);
252 return PARSE_INCOMPLETE;
253 }
254 } else if (is_separator_char(c)) {
255 /* scan separator */
256 token->buf = cursor++;
257 token_type = TT_SEPARATOR;
258 token->length = (size_t)1;
259 } else if (is_control_char(c)) {
260 /* scan ctrl char */
261 token->buf = cursor++;
262 token_type = TT_CTRL;
263 token->length = (size_t)1;
264 } else
265 return PARSE_FAILURE;
266
267 scanner->cursor += token->length; /* move to next token */
268 *tok_type = token_type;
269 return PARSE_OK;
270}

324UPNP_INLINE parse_status_t skip_blank_lines(
325 scanner_t* scanner
326) {
327 memptr token;
328 token_type_t tok_type;
329 parse_status_t status;
330
331 /* skip ws, crlf */
332 do {
333 status = scanner_get_token(scanner, &token, &tok_type);
334 } while (status == (parse_status_t)PARSE_OK &&
335 (tok_type == (token_type_t)TT_WHITESPACE ||
336 tok_type == (token_type_t)TT_CRLF));
337 if (status == (parse_status_t)PARSE_OK) {
338 /* pushback a non-whitespace token */
339 scanner->cursor -= token.length;
340 }
341
342 return status;
343}

354UPNP_INLINE parse_status_t skip_lws(
355 scanner_t* scanner
356) {
357 memptr token;
358 token_type_t tok_type;
359 parse_status_t status;
360 size_t save_pos;
361 int matched;
362
363 do {
364 save_pos = scanner->cursor;
365 matched = 0;
366
367 /* get CRLF or WS */
368 status = scanner_get_token(scanner, &token, &tok_type);
369 if (status == (parse_status_t)PARSE_OK) {
370 if (tok_type == (token_type_t)TT_CRLF) {
371 /* get WS */
372 status = scanner_get_token(scanner, &token, &tok_type);
373 }
374
375 if (status == (parse_status_t)PARSE_OK &&
376 tok_type == (token_type_t)TT_WHITESPACE) {
377 matched = 1;
378 } else {
379 /* did not match LWS; pushback token(s) */
380 scanner->cursor = save_pos;
381 }
382 }
383 } while (matched);
384
385 /* if entire msg is loaded, ignore an 'incomplete' warning */
386 if (status == (parse_status_t)PARSE_INCOMPLETE &&
387 scanner->entire_msg_loaded) {
388 status = PARSE_OK;
389 }
390
391 return status;
392}

404UPNP_INLINE parse_status_t match_non_ws_string(
405 scanner_t* scanner,
406 memptr* str
407) {
408 memptr token{};
409 token_type_t tok_type;
410 parse_status_t status{};
411 int done = 0;
412 size_t save_cursor;
413
414 save_cursor = scanner->cursor;
415
416 str->length = (size_t)0;
417 str->buf = scanner_get_str(scanner); /* point to next char in input */
418
419 while (!done) {
420 status = scanner_get_token(scanner, &token, &tok_type);
421 if (status == (parse_status_t)PARSE_OK &&
422 tok_type != (token_type_t)TT_WHITESPACE &&
423 tok_type != (token_type_t)TT_CRLF) {
424 /* append non-ws token */
425 str->length += token.length;
426 } else {
427 done = 1;
428 }
429 }
430
431 if (status == (parse_status_t)PARSE_OK) {
432 /* last token was WS; push it back in */
433 scanner->cursor -= token.length;
434 }
435 /* tolerate 'incomplete' msg */
436 if (status == (parse_status_t)PARSE_OK ||
437 (status == (parse_status_t)PARSE_INCOMPLETE &&
438 scanner->entire_msg_loaded)) {
439 if (str->length == (size_t)0) {
440 /* no strings found */
441 return PARSE_NO_MATCH;
442 } else {
443 return PARSE_OK;
444 }
445 } else {
446 /* error -- pushback tokens */
447 scanner->cursor = save_cursor;
448 return status;
449 }
450}

464UPNP_INLINE parse_status_t match_raw_value(
465 scanner_t* scanner,
466 memptr* raw_value
467) {
468 memptr token;
469 token_type_t tok_type;
470 parse_status_t status{};
471 int done = 0;
472 int saw_crlf = 0;
473 size_t pos_at_crlf = (size_t)0;
474 size_t save_pos;
475 char c;
476
477 save_pos = scanner->cursor;
478
479 /* value points to start of input */
480 raw_value->buf = scanner_get_str(scanner);
481 raw_value->length = (size_t)0;
482
483 while (!done) {
484 status = scanner_get_token(scanner, &token, &tok_type);
485 if (status == (parse_status_t)PARSE_OK) {
486 if (!saw_crlf) {
487 if (tok_type == (token_type_t)TT_CRLF) {
488 /* CRLF could end value */
489 saw_crlf = 1;
490
491 /* save input position at start of CRLF
492 */
493 pos_at_crlf = scanner->cursor - token.length;
494 }
495 /* keep appending value */
496 raw_value->length += token.length;
497 } else /* already seen CRLF */
498 {
499 if (tok_type == (token_type_t)TT_WHITESPACE) {
500 /* start again; forget CRLF */
501 saw_crlf = 0;
502 raw_value->length += token.length;
503 } else {
504 /* non-ws means value ended just before
505 * CRLF */
506 done = 1;
507
508 /* point to the crlf which ended the
509 * value */
510 scanner->cursor = pos_at_crlf;
511 }
512 }
513 } else {
514 /* some kind of error; restore scanner position */
515 scanner->cursor = save_pos;
516 done = 1;
517 }
518 }
519
520 if (status == (parse_status_t)PARSE_OK) {
521 /* trim whitespace on right side of value */
522 while (raw_value->length > (size_t)0) {
523 /* get last char */
524 c = raw_value->buf[raw_value->length - (size_t)1];
525
526 if (c != ' ' && c != '\t' && c != TOKCHAR_CR && c != TOKCHAR_LF) {
527 /* done; no more whitespace */
528 break;
529 }
530 /* remove whitespace */
531 raw_value->length--;
532 }
533 }
534
535 return status;
536}

551UPNP_INLINE parse_status_t match_int(
552 scanner_t* scanner,
553 int base,
554 int* value
555) {
556 memptr token;
557 token_type_t tok_type;
558 parse_status_t status;
559 long num;
560 char* end_ptr;
561 size_t save_pos;
562
563 save_pos = scanner->cursor;
564 status = scanner_get_token(scanner, &token, &tok_type);
565 if (status == (parse_status_t)PARSE_OK) {
566 if (tok_type == (token_type_t)TT_IDENTIFIER) {
567 errno = 0;
568 num = strtol(token.buf, &end_ptr, base);
569 /* all and only those chars in token should be used for
570 * num */
571 if (num < 0 || end_ptr != token.buf + token.length ||
572 ((num == LONG_MIN || num == LONG_MAX) && (errno == ERANGE))) {
573 status = PARSE_NO_MATCH;
574 }
575 /* save result */
576 *value = (int)num;
577 } else {
578 /* token must be an identifier */
579 status = PARSE_NO_MATCH;
580 }
581 }
582 if (status != (parse_status_t)PARSE_OK) {
583 /* restore scanner position for bad values */
584 scanner->cursor = save_pos;
585 }
586
587 return status;
588}

602UPNP_INLINE parse_status_t read_until_crlf(
603 scanner_t* scanner,
604 memptr* str
605) {
606 memptr token;
607 token_type_t tok_type;
608 parse_status_t status;
609 size_t start_pos;
610
611 start_pos = scanner->cursor;
612 str->buf = scanner_get_str(scanner);
613
614 /* read until we hit a crlf */
615 do {
616 status = scanner_get_token(scanner, &token, &tok_type);
617 } while (status == (parse_status_t)PARSE_OK &&
618 tok_type != (token_type_t)TT_CRLF);
619
620 if (status == (parse_status_t)PARSE_OK) {
621 /* pushback crlf in stream */
622 scanner->cursor -= token.length;
623
624 /* str should include all strings except crlf at the end */
625 str->length = scanner->cursor - start_pos;
626 }
627
628 return status;
629}

642UPNP_INLINE parse_status_t match_char(
643 scanner_t* scanner,
644 char c,
645 int case_sensitive
647) {
648 char scan_char;
649
650 if (scanner->cursor >= scanner->msg->length) {
651 return PARSE_INCOMPLETE;
652 }
653 /* read next char from scanner */
654 scan_char = scanner->msg->buf[scanner->cursor++];
655
656 if (case_sensitive) {
657 return c == scan_char ? PARSE_OK : PARSE_NO_MATCH;
658 } else {
659 return tolower(c) == tolower(scan_char) ? PARSE_OK : PARSE_NO_MATCH;
660 }
661}

663/*
664 *
665 *
666 * args for ...
667 * %d, int * (31-bit positive integer)
668 * %x, int * (31-bit postive number encoded as hex)
669 * %s, memptr* (simple identifier)
670 * %q, memptr* (quoted string)
671 * %S, memptr* (non-whitespace string)
672 * %R, memptr* (raw value)
673 * %U, uri_type* (url)
674 * %L, memptr* (string until end of line)
675 * %P, int * (current index of the string being scanned)
676 *
677 * no args for
678 * ' ' LWS*
679 * \t whitespace
680 * "%%" matches '%'
681 * "% " matches ' '
682 * %c matches CRLF
683 * %i ignore case in literal matching
684 * %n case-sensitive matching in literals
685 * %w optional whitespace; (similar to '\t',
686 * except whitespace is optional)
687 * %0 (zero) match null-terminator char '\0'
688 * (can only be used as last char in fmt)
689 * use only in matchstr(), not match()
690 * other chars match literally
691 *
692 * returns:
693 * PARSE_OK
694 * PARSE_INCOMPLETE
695 * PARSE_FAILURE -- bad input
696 * PARSE_NO_MATCH -- input does not match pattern
697 */
698
712parse_status_t vfmatch( //
713 scanner_t* scanner,
714 const char* fmt,
715 va_list argp
716) {
717 char c;
718 const char* fmt_ptr = fmt;
719 parse_status_t status{};
720 memptr* str_ptr;
721 memptr temp_str;
722 int* int_ptr;
723 uri_type* uri_ptr;
724 size_t save_pos;
725 int stat;
726 int case_sensitive = 1;
727 memptr token;
728 token_type_t tok_type;
729 int base;
730
731 assert(scanner != NULL);
732 assert(fmt != NULL);
733
734 /* save scanner pos; to aid error recovery */
735 save_pos = scanner->cursor;
736
737 status = PARSE_OK;
738 while (((c = *fmt_ptr++) != 0) && (status == (parse_status_t)PARSE_OK)) {
739 if (c == '%') {
740 c = *fmt_ptr++;
741 switch (c) {
742 case 'R': /* raw value */
743 str_ptr = va_arg(argp, memptr*);
744 assert(str_ptr != NULL);
745 status = match_raw_value(scanner, str_ptr);
746 break;
747 case 's': /* simple identifier */
748 str_ptr = va_arg(argp, memptr*);
749 assert(str_ptr != NULL);
750 status = scanner_get_token(scanner, str_ptr, &tok_type);
751 if (status == (parse_status_t)PARSE_OK &&
752 tok_type != (token_type_t)TT_IDENTIFIER) {
753 /* not an identifier */
754 status = PARSE_NO_MATCH;
755 }
756 break;
757 case 'c': /* crlf */
758 status = scanner_get_token(scanner, &token, &tok_type);
759 if (status == (parse_status_t)PARSE_OK &&
760 tok_type != (token_type_t)TT_CRLF) {
761 /* not CRLF token */
762 status = PARSE_NO_MATCH;
763 }
764 break;
765 case 'd': /* integer */
766 case 'x': /* hex number */
767 int_ptr = va_arg(argp, int*);
768 assert(int_ptr != NULL);
769 base = c == 'd' ? 10 : 16;
770 status = match_int(scanner, base, int_ptr);
771 break;
772 case 'S': /* non-whitespace string */
773 case 'U': /* uri */
774 if (c == 'S') {
775 str_ptr = va_arg(argp, memptr*);
776 } else {
777 str_ptr = &temp_str;
778 }
779 assert(str_ptr != NULL);
780 status = match_non_ws_string(scanner, str_ptr);
781 if (c == 'U' && status == (parse_status_t)PARSE_OK) {
782 uri_ptr = va_arg(argp, uri_type*);
783 assert(uri_ptr != NULL);
784 stat = parse_uri(str_ptr->buf, str_ptr->length, uri_ptr);
785 if (stat != HTTP_SUCCESS) {
786 status = PARSE_NO_MATCH;
787 }
788 }
789 break;
790 case 'L': /* string till eol */
791 str_ptr = va_arg(argp, memptr*);
792 assert(str_ptr != NULL);
793 status = read_until_crlf(scanner, str_ptr);
794 break;
795 case ' ': /* match space */
796 case '%': /* match percentage symbol */
797 status = match_char(scanner, c, case_sensitive);
798 break;
799 case 'n': /* case-sensitive match */
800 case_sensitive = 1;
801 break;
802 case 'i': /* ignore case */
803 case_sensitive = 0;
804 break;
805 case 'q': /* quoted string */
806 str_ptr = (memptr*)va_arg(argp, memptr*);
807 status = scanner_get_token(scanner, str_ptr, &tok_type);
808 if (status == (parse_status_t)PARSE_OK &&
809 tok_type != (token_type_t)TT_QUOTEDSTRING) {
810 status = PARSE_NO_MATCH; /* not a quoted
811 string */
812 }
813 break;
814 case 'w':
815 /* optional whitespace */
816 status = scanner_get_token(scanner, &token, &tok_type);
817 if (status == (parse_status_t)PARSE_OK &&
818 tok_type != (token_type_t)TT_WHITESPACE) {
819 /* restore non-whitespace token */
820 scanner->cursor -= token.length;
821 }
822 break;
823 case 'P':
824 /* current pos of scanner */
825 int_ptr = va_arg(argp, int*);
826 assert(int_ptr != NULL);
827 *int_ptr = (int)scanner->cursor;
828 break;
829 /* valid only in matchstr() */
830 case '0':
831 /* end of msg? */
832 /* check that we are 1 beyond last char */
833 if (scanner->cursor == scanner->msg->length &&
834 scanner->msg->buf[scanner->cursor] == '\0') {
835 status = PARSE_OK;
836 } else {
837 status = PARSE_NO_MATCH;
838 }
839 break;
840 default:
841 /* unknown option */
842 assert(0);
843 }
844 } else {
845 switch (c) {
846 case ' ': /* LWS* */
847 status = skip_lws(scanner);
848 break;
849 case '\t': /* Whitespace */
850 status = scanner_get_token(scanner, &token, &tok_type);
851 if (status == (parse_status_t)PARSE_OK &&
852 tok_type != (token_type_t)TT_WHITESPACE) {
853 /* not whitespace token */
854 status = PARSE_NO_MATCH;
855 }
856 break;
857 default: /* match characters */
858 {
859 status = match_char(scanner, c, case_sensitive);
860 }
861 }
862 }
863 }
864 if (status != (parse_status_t)PARSE_OK) {
865 /* on error, restore original scanner pos */
866 scanner->cursor = save_pos;
867 }
868
869 return status;
870}

1358 memptr line;
1359 char save_char;
1360 int num_scanned;
1361 int i;
1362 size_t n;
1363 char* p;
1364
1365 assert(parser->position == POS_RESPONSE_LINE);
1366
1367 status = skip_blank_lines(&parser->scanner);
1368 if (status != (parse_status_t)PARSE_OK)
1369 return status;
1370 /* response line */
1371 /*status = match( &parser->scanner, "%ihttp%w/%w%d\t.\t%d\t%d\t%L%c", */
1372 /* &hmsg->major_version, &hmsg->minor_version, */
1373 /* &hmsg->status_code, &hmsg->status_msg ); */
1374 status = match(&parser->scanner, "%ihttp%w/%w%L%c", &line);
1375 if (status != (parse_status_t)PARSE_OK)
1376 return status;
1377 save_char = line.buf[line.length];
1378 line.buf[line.length] = '\0'; /* null-terminate */
1379 /* scan http version and ret code */
1380#ifdef _WIN32
1381 num_scanned = sscanf_s(line.buf,
1382#else
1383 num_scanned = sscanf(line.buf,
1384#endif
1385 "%d . %d %d", &hmsg->major_version,
1386 &hmsg->minor_version, &hmsg->status_code);
1387 line.buf[line.length] = save_char; /* restore */
1388 if (num_scanned != 3 || hmsg->major_version < 0 ||
1389 hmsg->minor_version < 0 || hmsg->status_code < 0)
1390 /* bad response line */
1391 return PARSE_FAILURE;
1392 /* point to status msg */
1393 p = line.buf;
1394 /* skip 3 ints */
1395 for (i = 0; i < 3; i++) {
1396 /* go to start of num */
1397 while (!isdigit(*p))
1398 p++;
1399 /* skip int */
1400 while (isdigit(*p))
1401 p++;
1402 }
1403 /* whitespace must exist after status code */
1404 if (*p != ' ' && *p != '\t')
1405 return PARSE_FAILURE;
1406 /* skip whitespace */
1407 while (*p == ' ' || *p == '\t')
1408 p++;
1409 /* now, p is at start of status msg */
1410 n = line.length - ((size_t)p - (size_t)line.buf);
1411 if (membuffer_assign(&hmsg->status_msg, p, n) != 0) {
1412 /* out of mem */
1413 parser->http_error_code = HTTP_INTERNAL_SERVER_ERROR;
1414 return PARSE_FAILURE;
1415 }
1416 parser->position = POS_HEADERS; /* move to headers */
1417
1418 return PARSE_OK;

1422parse_status_t parser_parse_headers(http_parser_t* parser) {
1423 parse_status_t status;
1424 memptr token;
1425 memptr hdr_value;
1426 token_type_t tok_type;
1427 scanner_t* scanner = &parser->scanner;
1428 size_t save_pos;
1429 http_header_t* header;
1430 int header_id;
1431 int ret = 0;
1432 int index;
1433 http_header_t* orig_header;
1434 char save_char;
1435 int ret2;
1436 static char zero = 0;
1437
1438 assert(parser->position == (parser_pos_t)POS_HEADERS ||
1439 parser->ent_position == ENTREAD_CHUNKY_HEADERS);
1440
1441 while (1) {
1442 save_pos = scanner->cursor;
1443 /* check end of headers */
1444 status = scanner_get_token(scanner, &token, &tok_type);
1445 if (status != (parse_status_t)PARSE_OK) {
1446 /* pushback tokens; useful only on INCOMPLETE error */
1447 scanner->cursor = save_pos;
1448 return status;
1449 }
1450 switch (tok_type) {
1451 case TT_CRLF:
1452 /* end of headers */
1453 if ((parser->msg.is_request) &&
1454 (parser->msg.method == (http_method_t)HTTPMETHOD_POST)) {
1455 parser->position = POS_COMPLETE; /*post entity parsing */
1456 /*is handled separately */
1457 return PARSE_SUCCESS;
1458 }
1459 parser->position = POS_ENTITY; /* read entity next */
1460 return PARSE_OK;
1461 case TT_IDENTIFIER:
1462 /* not end; read header */
1463 break;
1464 default:
1465 return PARSE_FAILURE; /* didn't see header name */
1466 }
1467 status = match(scanner, " : %R%c", &hdr_value);
1468 if (status != (parse_status_t)PARSE_OK) {
1469 /* pushback tokens; useful only on INCOMPLETE error */
1470 scanner->cursor = save_pos;
1471 return status;
1472 }
1473 /* add header */
1474 /* find header */
1476 index = map_str_to_int(token.buf, token.length, &Http_Header_Names[0],
1477 static_cast<int>(Http_Header_Names.size()), 0);
1478 if (index >= 0) {
1479 /*Check if it is a soap header */
1480 // No problem with type_cast to 'long unsigned int', index is
1481 // checked to be >= 0.
1482 if (Http_Header_Names[static_cast<size_t>(index)].id ==
1483 HDR_SOAPACTION) {
1484 parser->msg.method = SOAPMETHOD_POST;
1485 }
1486 // No problem with type_cast to 'long unsigned int', index is
1487 // checked to be >= 0.
1488 header_id = Http_Header_Names[static_cast<size_t>(index)].id;
1489 orig_header = httpmsg_find_hdr(&parser->msg, header_id, NULL);
1490 } else {
1491 header_id = HDR_UNKNOWN;
1492 save_char = token.buf[token.length];
1493 token.buf[token.length] = '\0';
1494 orig_header = httpmsg_find_hdr_str(&parser->msg, token.buf);
1495 token.buf[token.length] = save_char; /* restore */
1496 }
1497 if (orig_header == NULL) {
1498 /* add new header */
1499 header = (http_header_t*)malloc(sizeof(http_header_t));
1500 if (header == NULL) {
1501 parser->http_error_code = HTTP_INTERNAL_SERVER_ERROR;
1502 return PARSE_FAILURE;
1503 }
1504 membuffer_init(&header->name_buf);
1505 membuffer_init(&header->value);
1506 /* value can be 0 length */
1507 if (hdr_value.length == (size_t)0) {
1508 hdr_value.buf = &zero;
1509 hdr_value.length = (size_t)1;
1510 }
1511 /* save in header in buffers */
1512 if (membuffer_assign(&header->name_buf, token.buf, token.length) ||
1513 membuffer_assign(&header->value, hdr_value.buf,
1514 hdr_value.length)) {
1515 /* not enough mem */
1516 membuffer_destroy(&header->value);
1517 membuffer_destroy(&header->name_buf);
1518 free(header);
1519 parser->http_error_code = HTTP_INTERNAL_SERVER_ERROR;
1520 return PARSE_FAILURE;
1521 }
1522 header->name.buf = header->name_buf.buf;
1523 header->name.length = header->name_buf.length;
1524 header->name_id = header_id;
1525 if (!ListAddTail(&parser->msg.headers, header)) {
1526 membuffer_destroy(&header->value);
1527 membuffer_destroy(&header->name_buf);
1528 free(header);
1529 parser->http_error_code = HTTP_INTERNAL_SERVER_ERROR;
1530 return PARSE_FAILURE;
1531 }
1532 } else if (hdr_value.length > (size_t)0) {
1533 /* append value to existing header */
1534 /* append space */
1535 ret = membuffer_append_str(&orig_header->value, ", ");
1536 /* append continuation of header value */
1537 ret2 = membuffer_append(&orig_header->value, hdr_value.buf,
1538 hdr_value.length);
1539 if (ret == UPNP_E_OUTOF_MEMORY || ret2 == UPNP_E_OUTOF_MEMORY) {
1540 /* not enuf mem */
1541 parser->http_error_code = HTTP_INTERNAL_SERVER_ERROR;
1542 return PARSE_FAILURE;
1543 }
1544 }
1545 } /* end while */
1546}

1627 }

1630 return PARSE_CONTINUE_1;
1631}
