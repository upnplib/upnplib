66int GeneratePropertySet(
68 char** names,
70 char** values,
72 int count,
74 DOMString* out) {
75 char* buffer;
76 int counter = 0;
77 size_t size = 0;
78
79 /*size += strlen(XML_VERSION);*/
80 size += strlen(XML_PROPERTYSET_HEADER);
81 size += strlen("</e:propertyset>\n\n");
82 for (counter = 0; counter < count; counter++) {
83 size += strlen("<e:property>\n</e:property>\n");
84 size += 2 * strlen(names[counter]) + strlen(values[counter]) +
85 strlen("<></>\n");
86 }
87
88 size_t buffer_len{size + 1};
89 buffer = (char*)malloc(buffer_len);
90 if (buffer == NULL)
91 return UPNP_E_OUTOF_MEMORY;
92 memset(buffer, 0, size + 1);
93 /*
94 strcpy(buffer,XML_VERSION);
95 strcat(buffer, XML_PROPERTYSET_HEADER);
96 */
97 strcpy(buffer, XML_PROPERTYSET_HEADER);
98 for (counter = 0; counter < count; counter++) {
99 strcat(buffer, "<e:property>\n");
100 snprintf(&buffer[strlen(buffer)], buffer_len,
101 "<%s>%s</%s>\n</e:property>\n", names[counter],
102 values[counter], names[counter]);
103 }
104 strcat(buffer, "</e:propertyset>\n\n");
105 *out = ixmlCloneDOMString(buffer);
106 free(buffer);
107
108 return XML_SUCCESS;

136UPNP_INLINE int notify_send_and_recv(
138 uri_type* destination_url,
140 membuffer* mid_msg,
142 char* propertySet,
144 http_parser_t* response) {
145 uri_type url;
146 SOCKET conn_fd;
147 membuffer start_msg;
148 int ret_code;
149 int err_code;
150 int timeout;
151 SOCKINFO info;
152 const char* CRLF = "\r\n";
153
154 /* connect */
155 UpnpPrintf(UPNP_ALL, GENA, __FILE__, __LINE__, "gena notify to: %.*s\n",
156 (int)destination_url->hostport.text.size,
157 destination_url->hostport.text.buff);
158
159 conn_fd = http_Connect(destination_url, &url);
160 if (conn_fd < 0)
161 /* return UPNP error */
162 return UPNP_E_SOCKET_CONNECT;
163 ret_code = sock_init(&info, conn_fd);
164 if (ret_code) {
165 sock_destroy(&info, SD_BOTH);
166 return ret_code;
167 }
168 /* make start line and HOST header */
169 membuffer_init(&start_msg);
170 if (http_MakeMessage(&start_msg, 1, 1,
171 "q"
172 "s",
173 HTTPMETHOD_NOTIFY, &url, mid_msg->buf) != 0) {
174 membuffer_destroy(&start_msg);
175 sock_destroy(&info, SD_BOTH);
176 return UPNP_E_OUTOF_MEMORY;
177 }
178 timeout = GENA_NOTIFICATION_SENDING_TIMEOUT;
179 /* send msg (note: end of notification will contain "\r\n" twice) */
180 ret_code = http_SendMessage(&info, &timeout, "bbb", start_msg.buf,
181 start_msg.length, propertySet,
182 strlen(propertySet), CRLF, strlen(CRLF));
183 if (ret_code) {
184 membuffer_destroy(&start_msg);
185 sock_destroy(&info, SD_BOTH);
186 return ret_code;
187 }
188 timeout = GENA_NOTIFICATION_ANSWERING_TIMEOUT;
189 ret_code = http_RecvMessage(&info, response, HTTPMETHOD_NOTIFY, &timeout,
190 &err_code);
191 if (ret_code) {
192 membuffer_destroy(&start_msg);
193 sock_destroy(&info, SD_BOTH);
194 httpmsg_destroy(&response->msg);
195 return ret_code;
196 }
197 /* should shutdown completely when closing socket */
198 sock_destroy(&info, SD_BOTH);
199 membuffer_destroy(&start_msg);
200
201 return UPNP_E_SUCCESS;
202}

215int genaNotify(
218 char* headers,
220 char* propertySet,
223 subscription* sub) {
224 size_t i;
225 membuffer mid_msg;
226 uri_type* url;
227 http_parser_t response{};
228 int return_code = -1;
229
230 membuffer_init(&mid_msg);
231 if (http_MakeMessage(&mid_msg, 1, 1,
232 "s"
233 "ssc"
234 "sdcc",
235 headers, "SID: ", sub->sid,
236 "SEQ: ", sub->ToSendEventKey) != 0) {
237 membuffer_destroy(&mid_msg);
238 return UPNP_E_OUTOF_MEMORY;
239 }
240 /* send a notify to each url until one goes thru */
241 for (i = 0; i < sub->DeliveryURLs.size; i++) {
242 url = &sub->DeliveryURLs.parsedURLs[i];
243 return_code =
244 notify_send_and_recv(url, &mid_msg, propertySet, &response);
245 if (return_code == UPNP_E_SUCCESS)
246 break;
247 }
248 membuffer_destroy(&mid_msg);
249 if (return_code == UPNP_E_SUCCESS) {
250 if (response.msg.status_code == HTTP_OK)
251 return_code = GENA_SUCCESS;
252 else {
253 if (response.msg.status_code == HTTP_PRECONDITION_FAILED)
254 /*Invalid SID gets removed */
255 return_code = GENA_E_NOTIFY_UNACCEPTED_REMOVE_SUB;
256 else
257 return_code = GENA_E_NOTIFY_UNACCEPTED;
258 }
259 httpmsg_destroy(&response.msg);
260 }
261
262 return return_code;
263}

273void genaNotifyThread(
276 void* input) {
277 subscription* sub;
278 service_info* service;
279 subscription sub_copy;
280 notify_thread_struct* in = (notify_thread_struct*)input;
281 int return_code;
282 struct Handle_Info* handle_info;
283
284 /* This should be a HandleLock and not a HandleReadLock otherwise if
285 * there is a lot of notifications, then multiple threads will acquire a
286 * read lock and the thread which sends the notification will be blocked
287 * forever on the HandleLock at the end of this function. */
288 /*HandleReadLock(); */
289 HandleLock();
290 /* validate context */
291
292 if (GetHandleInfo(in->device_handle, &handle_info) != HND_DEVICE) {
293 free_notify_struct(in);
294 HandleUnlock();
295 return;
296 }
297
298 if (((service = FindServiceId(&handle_info->ServiceTable, in->servId,
299 in->UDN)) == 0) ||
300 !service->active ||
301 ((sub = GetSubscriptionSID(in->sid, service)) == 0) ||
302 copy_subscription(sub, &sub_copy) != HTTP_SUCCESS) {
303 free_notify_struct(in);
304 HandleUnlock();
305 return;
306 }
307
308 HandleUnlock();
309
310 /* send the notify */
311 return_code = genaNotify(in->headers, in->propertySet, &sub_copy);
312 freeSubscription(&sub_copy);
313 HandleLock();
314 if (GetHandleInfo(in->device_handle, &handle_info) != HND_DEVICE) {
315 free_notify_struct(in);
316 HandleUnlock();
317 return;
318 }
319 /* validate context */
320 if (((service = FindServiceId(&handle_info->ServiceTable, in->servId,
321 in->UDN)) == 0) ||
322 !service->active ||
323 ((sub = GetSubscriptionSID(in->sid, service)) == 0)) {
324 free_notify_struct(in);
325 HandleUnlock();
326 return;
327 }
328 sub->ToSendEventKey++;
329 if (sub->ToSendEventKey < 0)
330 /* wrap to 1 for overflow */
331 sub->ToSendEventKey = 1;
332
333 /* Remove head of event queue. Possibly activate next */
334 {
335 ListNode* node = ListHead(&sub->outgoing);
336 if (node)
337 ListDelNode(&sub->outgoing, node, 1);
338 if (ListSize(&sub->outgoing) > 0) {
339 ThreadPoolJob* job;
340 ListNode* node2 = ListHead(&sub->outgoing);
341 job = (ThreadPoolJob*)node2->item;
342 /* The new head of queue should not have already been
343 added to the pool, else something is very wrong */
344 assert(job->jobId != STALE_JOBID);
345
346 ThreadPoolAdd(&gSendThreadPool, job, NULL);
347 job->jobId = STALE_JOBID;
348 }
349 }
350
351 if (return_code == GENA_E_NOTIFY_UNACCEPTED_REMOVE_SUB)
352 RemoveSubscriptionSID(in->sid, service);
353 free_notify_struct(in);
354
355 HandleUnlock();
356}

366char* AllocGenaHeaders(
368 const DOMString propertySet) {
369 static const char* HEADER_LINE_1 =
370 "CONTENT-TYPE: text/xml; charset=\"utf-8\"\r\n";
371 static const char* HEADER_LINE_2A = "CONTENT-LENGTH: ";
372 static const char* HEADER_LINE_2B = "\r\n";
373 static const char* HEADER_LINE_3 = "NT: upnp:event\r\n";
374 static const char* HEADER_LINE_4 = "NTS: upnp:propchange\r\n";
375 char* headers = NULL;
376 size_t headers_size = 0;
377 int line = 0;
378 int rc = 0;
379
380 headers_size = strlen(HEADER_LINE_1) + strlen(HEADER_LINE_2A) +
381 MAX_CONTENT_LENGTH + strlen(HEADER_LINE_2B) +
382 strlen(HEADER_LINE_3) + strlen(HEADER_LINE_4) + 1;
383 headers = (char*)malloc(headers_size);
384 if (headers == NULL) {
385 line = __LINE__;
386 goto ExitFunction;
387 }
388 rc = snprintf(headers, headers_size, "%s%s%" PRIzu "%s%s%s", HEADER_LINE_1,
389 HEADER_LINE_2A, (unsigned long)strlen(propertySet) + 2,
390 HEADER_LINE_2B, HEADER_LINE_3, HEADER_LINE_4);
391
392ExitFunction:
393 if (headers == NULL || rc < 0 || (unsigned int)rc >= headers_size) {
394 UpnpPrintf(UPNP_ALL, GENA, __FILE__, line,
395 "AllocGenaHeaders(): Error UPNP_E_OUTOF_MEMORY\n");
396 }
397 return headers;
398}

401int genaInitNotifyCommon(UpnpDevice_Handle device_handle, char* UDN,
402 char* servId, DOMString propertySet,
403 const Upnp_SID sid) {
404 int ret = GENA_SUCCESS;
405 int line = 0;
406
407 int* reference_count = NULL;
408 char* UDN_copy = NULL;
409 char* servId_copy = NULL;
410 char* headers = NULL;
411 notify_thread_struct* thread_struct = NULL;
412
413 subscription* sub = NULL;
414 service_info* service = NULL;
415 struct Handle_Info* handle_info;
416 ThreadPoolJob* job = NULL;
417
418 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__,
419 "GENA BEGIN INITIAL NOTIFY COMMON\n");
420
421 job = (ThreadPoolJob*)malloc(sizeof(ThreadPoolJob));
422 if (job == NULL) {
423 line = __LINE__;
424 ret = UPNP_E_OUTOF_MEMORY;
425 goto ExitFunction;
426 }
427 memset(job, 0, sizeof(ThreadPoolJob));
428
429 reference_count = (int*)malloc(sizeof(int));
430 if (reference_count == NULL) {
431 line = __LINE__;
432 ret = UPNP_E_OUTOF_MEMORY;
433 goto ExitFunction;
434 }
435 *reference_count = 0;
436
437 UDN_copy = strdup(UDN);
438 if (UDN_copy == NULL) {
439 line = __LINE__;
440 ret = UPNP_E_OUTOF_MEMORY;
441 goto ExitFunction;
442 }
443
444 servId_copy = strdup(servId);
445 if (servId_copy == NULL) {
446 line = __LINE__;
447 ret = UPNP_E_OUTOF_MEMORY;
448 goto ExitFunction;
449 }
450
451 HandleLock();
452
453 if (GetHandleInfo(device_handle, &handle_info) != HND_DEVICE) {
454 line = __LINE__;
455 ret = GENA_E_BAD_HANDLE;
456 goto ExitFunction;
457 }
458
459 service = FindServiceId(&handle_info->ServiceTable, servId, UDN);
460 if (service == NULL) {
461 line = __LINE__;
462 ret = GENA_E_BAD_SERVICE;
463 goto ExitFunction;
464 }
465 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__,
466 "FOUND SERVICE IN INIT NOTFY: UDN %s, ServID: %s", UDN, servId);
467
468 sub = GetSubscriptionSID(sid, service);
469 if (sub == NULL || sub->active) {
470 line = __LINE__;
471 ret = GENA_E_BAD_SID;
472 goto ExitFunction;
473 }
474 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__,
475 "FOUND SUBSCRIPTION IN INIT NOTIFY: SID %s", sid);
476 sub->active = 1;
477
478 headers = AllocGenaHeaders(propertySet);
479 if (headers == NULL) {
480 line = __LINE__;
481 ret = UPNP_E_OUTOF_MEMORY;
482 goto ExitFunction;
483 }
484
485 /* schedule thread for initial notification */
486
487 thread_struct = (notify_thread_struct*)malloc(sizeof(notify_thread_struct));
488 if (thread_struct == NULL) {
489 line = __LINE__;
490 ret = UPNP_E_OUTOF_MEMORY;
491 } else {
492 *reference_count = 1;
493 thread_struct->servId = servId_copy;
494 thread_struct->UDN = UDN_copy;
495 thread_struct->headers = headers;
496 thread_struct->propertySet = propertySet;
497 memset(thread_struct->sid, 0, sizeof(thread_struct->sid));
498 strncpy(thread_struct->sid, sid, sizeof(thread_struct->sid) - 1);
499 thread_struct->ctime = time(0);
500 thread_struct->reference_count = reference_count;
501 thread_struct->device_handle = device_handle;
502
503 TPJobInit(job, (start_routine)genaNotifyThread, thread_struct);
504 TPJobSetFreeFunction(job, (free_routine)free_notify_struct);
505 TPJobSetPriority(job, MED_PRIORITY);
506
507 ret = ThreadPoolAdd(&gSendThreadPool, job, NULL);
508 if (ret != 0) {
509 if (ret == EOUTOFMEM) {
510 line = __LINE__;
511 ret = UPNP_E_OUTOF_MEMORY;
512 }
513 } else {
514 ListNode* node = ListAddTail(&sub->outgoing, job);
515 if (node != NULL) {
516 ((ThreadPoolJob*)node->item)->jobId = STALE_JOBID;
517 line = __LINE__;
518 ret = GENA_SUCCESS;
519 } else {
520 line = __LINE__;
521 ret = UPNP_E_OUTOF_MEMORY;
522 }
523 }
524 }
525
526ExitFunction:
527 if (ret != GENA_SUCCESS) {
528 free(job);
529 free(thread_struct);
530 free(headers);
531 ixmlFreeDOMString(propertySet);
532 free(servId_copy);
533 free(UDN_copy);
534 free(reference_count);
535 }
536
537 HandleUnlock();
538
539 UpnpPrintf(UPNP_INFO, GENA, __FILE__, line,
540 "GENA END INITIAL NOTIFY COMMON, ret = %d\n", ret);
541
542 return ret;
543}

554void maybeDiscardEvents(LinkedList* listp) {
555 time_t now = time(0L);
556 notify_thread_struct* ntsp;
557
558 while (ListSize(listp) > 1) {
559 ListNode* node = ListHead(listp);
560 /* The first candidate is the second event: first non-active */
561 if (node == 0 || (node = node->next) == 0) {
562 /* Major inconsistency, really, should abort here. */
563 fprintf(stderr, "gena_device: maybeDiscardEvents: "
564 "list is inconsistent\n");
565 break;
566 }
567
568 ntsp = (notify_thread_struct*)(((ThreadPoolJob*)node->item)->arg);
569 if (ListSize(listp) > g_UpnpSdkEQMaxLen ||
570 now - ntsp->ctime > g_UpnpSdkEQMaxAge) {
571 free_notify_struct(ntsp);
572 free(node->item);
573 ListDelNode(listp, node, 0);
574 } else {
575 /* If the list is smaller than the max and the oldest
576 * task is young enough, stop pruning */
577 break;
578 }
579 }
580}

583int genaNotifyAllCommon(UpnpDevice_Handle device_handle, char* UDN,
584 char* servId, DOMString propertySet) {
585 int ret = GENA_SUCCESS;
586 int line = 0;
587
588 int* reference_count = NULL;
589 char* UDN_copy = NULL;
590 char* servId_copy = NULL;
591 char* headers = NULL;
592 notify_thread_struct* thread_s = NULL;
593
594 subscription* finger = NULL;
595 service_info* service = NULL;
596 struct Handle_Info* handle_info;
597
598 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__,
599 "GENA BEGIN NOTIFY ALL COMMON\n");
600
601 /* Keep this allocation first */
602 reference_count = (int*)malloc(sizeof(int));
603 if (reference_count == NULL) {
604 line = __LINE__;
605 ret = UPNP_E_OUTOF_MEMORY;
606 goto ExitFunction;
607 }
608 *reference_count = 0;
609
610 UDN_copy = strdup(UDN);
611 if (UDN_copy == NULL) {
612 line = __LINE__;
613 ret = UPNP_E_OUTOF_MEMORY;
614 goto ExitFunction;
615 }
616
617 servId_copy = strdup(servId);
618 if (servId_copy == NULL) {
619 line = __LINE__;
620 ret = UPNP_E_OUTOF_MEMORY;
621 goto ExitFunction;
622 }
623
624 headers = AllocGenaHeaders(propertySet);
625 if (headers == NULL) {
626 line = __LINE__;
627 ret = UPNP_E_OUTOF_MEMORY;
628 goto ExitFunction;
629 }
630
631 HandleLock();
632
633 if (GetHandleInfo(device_handle, &handle_info) != HND_DEVICE) {
634 line = __LINE__;
635 ret = GENA_E_BAD_HANDLE;
636 } else {
637 service = FindServiceId(&handle_info->ServiceTable, servId, UDN);
638 if (service != NULL) {
639 finger = GetFirstSubscription(service);
640 while (finger) {
641 ThreadPoolJob* job = NULL;
642 ListNode* node;
643
644 thread_s =
645 (notify_thread_struct*)malloc(sizeof(notify_thread_struct));
646 if (thread_s == NULL) {
647 line = __LINE__;
648 ret = UPNP_E_OUTOF_MEMORY;
649 break;
650 }
651
652 (*reference_count)++;
653 thread_s->reference_count = reference_count;
654 thread_s->UDN = UDN_copy;
655 thread_s->servId = servId_copy;
656 thread_s->headers = headers;
657 thread_s->propertySet = propertySet;
658 strncpy(thread_s->sid, finger->sid, sizeof thread_s->sid);
659 thread_s->sid[sizeof thread_s->sid - 1] = 0;
660 thread_s->ctime = time(0);
661 thread_s->device_handle = device_handle;
662
663 maybeDiscardEvents(&finger->outgoing);
664 job = (ThreadPoolJob*)malloc(sizeof(ThreadPoolJob));
665 if (!job) {
666 free(thread_s);
667 line = __LINE__;
668 ret = UPNP_E_OUTOF_MEMORY;
669 break;
670 }
671 memset(job, 0, sizeof(ThreadPoolJob));
672 TPJobInit(job, (start_routine)genaNotifyThread, thread_s);
673 TPJobSetFreeFunction(job, (free_routine)free_notify_struct);
674 TPJobSetPriority(job, MED_PRIORITY);
675 node = ListAddTail(&finger->outgoing, job);
676
677 /* If there is only one element on the list
678 (which we just
679 added), need to kickstart the threadpool */
680 if (ListSize(&finger->outgoing) == 1) {
681 ret = ThreadPoolAdd(&gSendThreadPool, job, NULL);
682 if (ret != 0) {
683 line = __LINE__;
684 if (ret == EOUTOFMEM) {
685 line = __LINE__;
686 ret = UPNP_E_OUTOF_MEMORY;
687 }
688 break;
689 }
690 if (node) {
691 ((ThreadPoolJob*)(node->item))->jobId = STALE_JOBID;
692 }
693 }
694 finger = GetNextSubscription(service, finger);
695 }
696 } else {
697 line = __LINE__;
698 ret = GENA_E_BAD_SERVICE;
699 }
700 }
701
702ExitFunction:
703 /* The only case where we want to free memory here is if the
704 struct was never queued. Else, let the normal cleanup take place.
705 reference_count is allocated first so it's ok to do nothing if it's 0
706 */
707 if (reference_count && *reference_count == 0) {
708 free(headers);
709 ixmlFreeDOMString(propertySet);
710 free(servId_copy);
711 free(UDN_copy);
712 free(reference_count);
713 }
714
715 HandleUnlock();
716
717 UpnpPrintf(UPNP_INFO, GENA, __FILE__, line,
718 "GENA END NOTIFY ALL COMMON, ret = %d\n", ret);
719
720 return ret;
721}

779 response.length);
780
781 membuffer_destroy(&response);
782

784}

1057int gena_validate_delivery_urls(
1059 SOCKINFO* info,
1061 URL_list* url_list) {
1062 size_t i = 0;
1063 struct in_addr genaAddr4;
1064 struct in_addr genaNetmask;
1065 struct sockaddr_in* deliveryAddr4 = NULL;
1066 struct in6_addr genaAddr6Lla;
1067 struct in6_addr genaAddr6UlaGua;
1068 struct in6_addr* genaAddr6 = NULL;
1069 unsigned int if_prefix;
1070 struct sockaddr_in6* deliveryAddr6 = NULL;
1071 char deliveryAddrString[INET6_ADDRSTRLEN];
1072
1073 if (info == NULL || url_list == NULL) {
1074 return 0;
1075 }
1076
1077 switch (info->foreign_sockaddr.ss_family) {
1078 case AF_INET:
1079 if (!inet_pton(AF_INET, gIF_IPV4, &genaAddr4)) {
1080 return -1;
1081 }
1082
1083 if (!inet_pton(AF_INET, gIF_IPV4_NETMASK, &genaNetmask)) {
1084 return -1;
1085 }
1086
1087 for (i = 0; i < url_list->size; i++) {
1088 deliveryAddr4 = (struct sockaddr_in*)&url_list->parsedURLs[i]
1089 .hostport.IPaddress;
1090 if ((deliveryAddr4->sin_addr.s_addr & genaNetmask.s_addr) !=
1091 (genaAddr4.s_addr & genaNetmask.s_addr)) {
1092 inet_ntop(AF_INET, &deliveryAddr4->sin_addr, deliveryAddrString,
1093 sizeof(deliveryAddrString));
1094 UpnpPrintf(UPNP_CRITICAL, GENA, __FILE__, __LINE__,
1095 "DeliveryURL %s is invalid.\n"
1096 "It is not in the expected network "
1097 "segment (IPv4: %s, netmask: %s)\n",
1098 deliveryAddrString, gIF_IPV4, gIF_IPV4_NETMASK);
1099 return -1;
1100 }
1101 }
1102 break;
1103 case AF_INET6:
1104 if (!inet_pton(AF_INET6, gIF_IPV6, &genaAddr6Lla)) {
1105 return -1;
1106 }
1107
1108 if (!inet_pton(AF_INET6, gIF_IPV6_ULA_GUA, &genaAddr6UlaGua)) {
1109 return -1;
1110 }
1111
1112 for (i = 0; i < url_list->size; i++) {
1113 deliveryAddr6 = (struct sockaddr_in6*)&url_list->parsedURLs[i]
1114 .hostport.IPaddress;
1115 if (IN6_IS_ADDR_LINKLOCAL(&deliveryAddr6->sin6_addr)) {
1116 genaAddr6 = &genaAddr6Lla;
1117 if_prefix = gIF_IPV6_PREFIX_LENGTH;
1118 } else {
1119 genaAddr6 = &genaAddr6UlaGua;
1120 if_prefix = gIF_IPV6_ULA_GUA_PREFIX_LENGTH;
1121 }
1122 /* We assume that IPv6 prefix is a multiple of 8 */
1123 if (memcmp(deliveryAddr6->sin6_addr.s6_addr, genaAddr6->s6_addr,
1124 if_prefix / 8)) {
1125 inet_ntop(AF_INET6, &deliveryAddr6->sin6_addr,
1126 deliveryAddrString, sizeof(deliveryAddrString));
1127 UpnpPrintf(UPNP_CRITICAL, GENA, __FILE__, __LINE__,
1128 "DeliveryURL %s is invalid.\n"
1129 "It is not in the expected network "
1130 "segment (IPv6: %s, prefix: %d)\n",
1131 deliveryAddrString,
1132 IN6_IS_ADDR_LINKLOCAL(&deliveryAddr6->sin6_addr)
1133 ? gIF_IPV6
1134 : gIF_IPV6_ULA_GUA,
1135 if_prefix);
1136 return -1;
1137 }
1138 }
1139 break;
1140 }
1141 return 0;

1182 error_respond(info, HTTP_BAD_REQUEST, request);
1183 goto exit_function;

1188 if (event_url_path == NULL) {
1189 error_respond(info, HTTP_INTERNAL_SERVER_ERROR, request);
1190 goto exit_function;
1191 }

1193 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__,

1199 event_url_path, info->foreign_sockaddr.ss_family, &device_handle,
1200 &handle_info, &service) != HND_DEVICE) {
1201 free(event_url_path);
1202 error_respond(info, HTTP_INTERNAL_SERVER_ERROR, request);
1203 HandleUnlock();
1204 goto exit_function;
1205 }
1206 free(event_url_path);

1208 if (service == NULL || !service->active) {
1209 error_respond(info, HTTP_NOT_FOUND, request);
1210 HandleUnlock();
1211 goto exit_function;
1212 }
1213
1214 UpnpPrintf(UPNP_INFO, GENA, __FILE__, __LINE__,
1215 "Subscription Request: Number of Subscriptions already %d\n "
1216 "Max Subscriptions allowed: %d\n",
1217 service->TotalSubscriptions, handle_info->MaxSubscriptions);
1218
1219 /* too many subscriptions */
1220 if (handle_info->MaxSubscriptions != -1 &&
1221 service->TotalSubscriptions >= handle_info->MaxSubscriptions) {
1222 error_respond(info, HTTP_INTERNAL_SERVER_ERROR, request);
1223 HandleUnlock();
1224 goto exit_function;
1225 }
1226 /* generate new subscription */
1227 sub = (subscription*)malloc(sizeof(subscription));
1228 if (sub == NULL) {
1229 error_respond(info, HTTP_INTERNAL_SERVER_ERROR, request);
1230 HandleUnlock();
1231 goto exit_function;
1232 }
1233 sub->ToSendEventKey = 0;
1234 sub->active = 0;
1235 sub->next = NULL;
1236 sub->DeliveryURLs.size = 0;
1237 sub->DeliveryURLs.URLs = NULL;
1238 sub->DeliveryURLs.parsedURLs = NULL;
1239 if (ListInit(&sub->outgoing, 0, free) != 0) {
1240 error_respond(info, HTTP_INTERNAL_SERVER_ERROR, request);
1241 HandleUnlock();
1242 goto exit_function;
1243 }
1244
1245 /* check for valid callbacks */
1246 if (httpmsg_find_hdr(request, HDR_CALLBACK, &callback_hdr) == NULL) {
1247 error_respond(info, HTTP_PRECONDITION_FAILED, request);
1248 freeSubscriptionList(sub);
1249 HandleUnlock();
1250 goto exit_function;
1251 }
1252 return_code = create_url_list(&callback_hdr, &sub->DeliveryURLs);
1253 if (return_code == 0) {
1254 error_respond(info, HTTP_PRECONDITION_FAILED, request);
1255 freeSubscriptionList(sub);
1256 HandleUnlock();
1257 goto exit_function;
1258 }

1260 error_respond(info, HTTP_INTERNAL_SERVER_ERROR, request);
1261 freeSubscriptionList(sub);
1262 HandleUnlock();
1263 goto exit_function;
1264 }

1266 if (return_code != 0) {
1267 error_respond(info, HTTP_PRECONDITION_FAILED, request);
1268 freeSubscriptionList(sub);
1269 HandleUnlock();
1270 goto exit_function;
1271 }
1272 /* set the timeout */
1273 if (httpmsg_find_hdr(request, HDR_TIMEOUT, &timeout_hdr) != NULL) {
1274 if (matchstr(timeout_hdr.buf, timeout_hdr.length, "%iSecond-%d%0",
1275 &time_out) == PARSE_OK) {
1276 /* nothing */
1277 } else if (memptr_cmp_nocase(&timeout_hdr, "Second-infinite") == 0) {
1278 /* infinite timeout */
1279 time_out = -1;
1280 } else {
1281 /* default is > 1800 seconds */
1282 time_out = DEFAULT_TIMEOUT;
1283 }
1284 }
1285 /* replace infinite timeout with max timeout, if possible */
1286 if (handle_info->MaxSubscriptionTimeOut != -1) {
1287 if (time_out == -1 || time_out > handle_info->MaxSubscriptionTimeOut) {
1288 time_out = handle_info->MaxSubscriptionTimeOut;
1289 }
1290 }
1291 if (time_out >= 0) {
1292 sub->expireTime = time(NULL) + time_out;
1293 } else {
1294 /* infinite time */
1295 sub->expireTime = 0;
1296 }
1297
1298 /* generate SID */
1299 uuid_create(&uid);
1300 upnp_uuid_unpack(&uid, temp_sid);
1301 rc = snprintf(sub->sid, sizeof(sub->sid), "uuid:%s", temp_sid);
1302
1303 /* respond OK */
1304 if (rc < 0 || (unsigned int)rc >= sizeof(sub->sid) ||
1305 (respond_ok(info, time_out, sub, request) != UPNP_E_SUCCESS)) {
1306 freeSubscriptionList(sub);
1307 HandleUnlock();
1308 goto exit_function;
1309 }
1310 /* add to subscription list */
1311 sub->next = service->subscriptionList;
1312 service->subscriptionList = sub;
1313 service->TotalSubscriptions++;
1314
1315 /* finally generate callback for init table dump */
1316 UpnpSubscriptionRequest_strcpy_ServiceId(request_struct,
1317 service->serviceId);
1318 UpnpSubscriptionRequest_strcpy_UDN(request_struct, service->UDN);
1319 UpnpSubscriptionRequest_strcpy_SID(request_struct, sub->sid);
1320
1321 /* copy callback */
1322 callback_fun = handle_info->Callback;
1323 cookie = handle_info->Cookie;
1324
1325 HandleUnlock();
1326
1327 /* make call back with request struct */
1328 /* in the future should find a way of mainting that the handle */
1329 /* is not unregistered in the middle of a callback */
1330 callback_fun(UPNP_EVENT_SUBSCRIPTION_REQUEST, request_struct, cookie);
1331
1332exit_function:
1333 UpnpSubscriptionRequest_delete(request_struct);
1334}
