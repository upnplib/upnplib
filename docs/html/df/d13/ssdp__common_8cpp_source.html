68void free_ssdp_event_handler_data(
71 void* the_data) {
72 ssdp_thread_data* data = (ssdp_thread_data*)the_data;
73
74 if (data != NULL) {
75 http_message_t* hmsg = &data->parser.msg;
76 /* free data */
77 httpmsg_destroy(hmsg);
78 free(data);
79 }

87inline int valid_ssdp_msg(
90 http_message_t* hmsg) {
91 memptr hdr_value;
92
93 /* check for valid methods - NOTIFY or M-SEARCH */
94 if (hmsg->method != (http_method_t)HTTPMETHOD_NOTIFY &&
95 hmsg->method != (http_method_t)HTTPMETHOD_MSEARCH &&
96 hmsg->request_method != (http_method_t)HTTPMETHOD_MSEARCH) {
97 return 0;
98 }
99 if (hmsg->request_method != (http_method_t)HTTPMETHOD_MSEARCH) {
100 /* check PATH == "*" */
101 if (hmsg->uri.type != (enum uriType)RELATIVE ||
102 strncmp("*", hmsg->uri.pathquery.buff, hmsg->uri.pathquery.size) !=
103 0) {
104 return 0;
105 }
106 /* check HOST header */
107 if (httpmsg_find_hdr(hmsg, HDR_HOST, &hdr_value) == NULL ||
108 (memptr_cmp(&hdr_value, "239.255.255.250:1900") != 0 &&
109 memptr_cmp(&hdr_value, "[FF02::C]:1900") != 0 &&
110 memptr_cmp(&hdr_value, "[ff02::c]:1900") != 0 &&
111 memptr_cmp(&hdr_value, "[FF05::C]:1900") != 0 &&
112 memptr_cmp(&hdr_value, "[ff05::c]:1900") != 0)) {
113 UpnpPrintf(UPNP_INFO, SSDP, __FILE__, __LINE__,
114 "Invalid HOST header from SSDP message\n");
115
116 return 0;
117 }
118 }
119
120 /* passed quick check */
121 return 1;
122}

132inline int start_event_handler(
135 void* Data) {
136 http_parser_t* parser = NULL;
137 parse_status_t status;
138 ssdp_thread_data* data = (ssdp_thread_data*)Data;
139
140 parser = &data->parser;
141 status = parser_parse(parser);
142 if (status == (parse_status_t)PARSE_FAILURE) {
143 if (parser->msg.method != (http_method_t)HTTPMETHOD_NOTIFY ||
144 !parser->valid_ssdp_notify_hack) {
145 UpnpPrintf(UPNP_INFO, SSDP, __FILE__, __LINE__,
146 "SSDP recvd bad msg code = %d\n", status);
147 /* ignore bad msg, or not enuf mem */
148 goto error_handler;
149 }
150 /* valid notify msg */
151 } else if (status != (parse_status_t)PARSE_SUCCESS) {
152 UpnpPrintf(UPNP_INFO, SSDP, __FILE__, __LINE__,
153 "SSDP recvd bad msg code = %d\n", status);
154
155 goto error_handler;
156 }
157 /* check msg */
158 if (valid_ssdp_msg(&parser->msg) != 1) {
159 goto error_handler;
160 }
161 /* done; thread will free 'data' */
162 return 0;
163
164error_handler:
165 free_ssdp_event_handler_data(data);
166 return -1;
167}

172void ssdp_event_handler_thread(
175 void* the_data) {
176 ssdp_thread_data* data = (ssdp_thread_data*)the_data;
177 http_message_t* hmsg = &data->parser.msg;
178
179 if (start_event_handler(the_data) != 0)
180 return;
181 /* send msg to device or ctrlpt */
182 if (hmsg->method == (http_method_t)HTTPMETHOD_NOTIFY ||
183 hmsg->request_method == (http_method_t)HTTPMETHOD_MSEARCH) {
185#ifdef COMPA_HAVE_CTRLPT_SSDP
186 ssdp_handle_ctrlpt_msg(hmsg, &data->dest_addr, 0);
187#endif
188 } else {
189#ifdef COMPA_HAVE_DEVICE_SSDP
190 ssdp_handle_device_request(hmsg, &data->dest_addr);
191#endif
192 }
193
194 /* free data */
195 free_ssdp_event_handler_data(data);
196}

209inline int create_ssdp_sock_v4(
211 SOCKET* ssdpSock) {
212 UPNPLIB_LOGINFO "MSG1075: Executing...\n";
213 int onOff;
214 u_char ttl = (u_char)4;
215 ip_mreq ssdpMcastAddr;
216 sockaddr_storage __ss;
217 sockaddr_in* ssdpAddr4 = (struct sockaddr_in*)&__ss;
218 int ret = 0;
219 in_addr addr;
220 upnplib::CSocketErr sockerrObj;
221
222 *ssdpSock = umock::sys_socket_h.socket(AF_INET, SOCK_DGRAM, 0);
223 if (*ssdpSock == INVALID_SOCKET) {
224 sockerrObj.catch_error();
225 UPNPLIB_LOGCRIT "MSG1076: Error in socket(): "
226 << sockerrObj.get_error_str() << ".\n";
227 return UPNP_E_OUTOF_SOCKET;
228 }
229 onOff = 1;
230 ret = umock::sys_socket_h.setsockopt(*ssdpSock, SOL_SOCKET, SO_REUSEADDR,
231 (char*)&onOff, sizeof(onOff));
232 if (ret == -1) {
233 sockerrObj.catch_error();
234 UPNPLIB_LOGCRIT "MSG1077: Error in setsockopt() SO_REUSEADDR: "
235 << sockerrObj.get_error_str() << ".\n";
236 ret = UPNP_E_SOCKET_ERROR;
237 goto error_handler;
238 }
239#if (defined(BSD) && !defined(__GNU__)) || defined(__APPLE__)
240 onOff = 1;
241 ret = umock::sys_socket_h.setsockopt(*ssdpSock, SOL_SOCKET, SO_REUSEPORT,
242 (char*)&onOff, sizeof(onOff));
243 if (ret == -1) {
244 sockerrObj.catch_error();
245 UPNPLIB_LOGCRIT "MSG1078: Error in setsockopt() SO_REUSEP: "
246 << sockerrObj.get_error_str() << ".\n";
247 ret = UPNP_E_SOCKET_ERROR;
248 goto error_handler;
249 }
250#endif /* BSD, __APPLE__ */
251 memset(&__ss, 0, sizeof(__ss));
252 ssdpAddr4->sin_family = (sa_family_t)AF_INET;
253 ssdpAddr4->sin_addr.s_addr = htonl(INADDR_ANY);
254 ssdpAddr4->sin_port = htons(SSDP_PORT);
255 ret = umock::sys_socket_h.bind(*ssdpSock, (sockaddr*)ssdpAddr4,
256 sizeof(*ssdpAddr4));
257 if (ret == -1) {
258 sockerrObj.catch_error();
259 UPNPLIB_LOGCRIT "MSG1079: Error in bind(), addr="
260 << INADDR_ANY << ", port=" << SSDP_PORT << ": "
261 << sockerrObj.get_error_str() << ".\n";
262 ret = UPNP_E_SOCKET_BIND;
263 goto error_handler;
264 }
265 /*
266 * See: https://man7.org/linux/man-pages/man7/ip.7.html
267 * Socket options, IP_ADD_MEMBERSHIP
268 *
269 * This memset actually sets imr_address to INADDR_ANY and
270 * imr_ifindex to zero, which make the system choose the appropriate
271 * interface.
272 *
273 * Still using "struct ip_mreq" instead of "struct ip_mreqn" because
274 * windows does not recognize the latter.
275 */
276 memset((void*)&ssdpMcastAddr, 0, sizeof ssdpMcastAddr);
277#ifdef _WIN32
278 inet_pton(AF_INET, (PCSTR)gIF_IPV4, &ssdpMcastAddr.imr_interface);
279 inet_pton(AF_INET, (PCSTR)SSDP_IP, &ssdpMcastAddr.imr_multiaddr);
280#else
281 ssdpMcastAddr.imr_interface.s_addr = inet_addr(gIF_IPV4);
282 /* ssdpMcastAddr.imr_address.s_addr = inet_addr(gIF_IPV4); */
283 ssdpMcastAddr.imr_multiaddr.s_addr = inet_addr(SSDP_IP);
284#endif
285 ret = umock::sys_socket_h.setsockopt(
286 *ssdpSock, IPPROTO_IP, IP_ADD_MEMBERSHIP, (char*)&ssdpMcastAddr,
287 sizeof(struct ip_mreq));
288 if (ret == -1) {
289 sockerrObj.catch_error();
290 UPNPLIB_LOGCRIT "MSG1080: Error in setsockopt() IP_ADD_MEMBERSHIP "
291 "(join multicast group): "
292 << sockerrObj.get_error_str() << ".\n";
293 ret = UPNP_E_SOCKET_ERROR;
294 goto error_handler;
295 }
296 /* Set multicast interface. */
297 memset((void*)&addr, 0, sizeof(struct in_addr));
298#ifdef _WIN32
299 inet_pton(AF_INET, (PCSTR)gIF_IPV4, &addr);
300#else
301 addr.s_addr = inet_addr(gIF_IPV4);
302#endif
303 ret = umock::sys_socket_h.setsockopt(*ssdpSock, IPPROTO_IP, IP_MULTICAST_IF,
304 (char*)&addr, sizeof addr);
305 if (ret == -1) {
306 sockerrObj.catch_error();
307 UPNPLIB_LOGINFO "MSG1081: Error in setsockopt() IP_MULTICAST_IF (set "
308 "multicast interface): "
309 << sockerrObj.get_error_str() << ".\n";
310 /* This is probably not a critical error, so let's continue. */
311 }
312 /* result is not checked becuase it will fail in WinMe and Win9x. */
313 umock::sys_socket_h.setsockopt(*ssdpSock, IPPROTO_IP, IP_MULTICAST_TTL,
314 (const char*)&ttl, sizeof(ttl));
315 onOff = 1;
316 ret = umock::sys_socket_h.setsockopt(*ssdpSock, SOL_SOCKET, SO_BROADCAST,
317 (char*)&onOff, sizeof(onOff));
318 if (ret == -1) {
319 sockerrObj.catch_error();
320 UPNPLIB_LOGCRIT
321 "MSG1082: Error in setsockopt() SO_BROADCAST (set broadcast): "
322 << sockerrObj.get_error_str() << ".\n";
323 ret = UPNP_E_NETWORK_ERROR;
324 goto error_handler;
325 }
326 ret = UPNP_E_SUCCESS;
327
328error_handler:
329 if (ret != UPNP_E_SUCCESS) {
330 UpnpCloseSocket(*ssdpSock);
331 }
332
333 return ret;
334}

592int unique_service_name(char* cmd, SsdpEvent* Evt) {
593 char TempBuf[LINE_SIZE - 3];
594 char* TempPtr = NULL;
595 char* Ptr = NULL;
596 char* ptr1 = NULL;
597 char* ptr2 = NULL;
598 char* ptr3 = NULL;
599 int CommandFound = 0;
600 size_t n = (size_t)0;
601
602 if (strstr(cmd, "uuid:schemas") != NULL) {
603 ptr1 = strstr(cmd, ":device");
604 if (ptr1 != NULL)
605 ptr2 = strstr(ptr1 + 1, ":");
606 else
607 return -1;
608 if (ptr2 != NULL)
609 ptr3 = strstr(ptr2 + 1, ":");
610 else
611 return -1;
612 if (ptr3 != NULL) {
613 if (strlen("uuid:") + strlen(ptr3 + 1) >= sizeof Evt->UDN)
614 return -1;
615 snprintf(Evt->UDN, sizeof Evt->UDN, "uuid:%s", ptr3 + 1);
616 } else
617 return -1;
618 ptr1 = strstr(cmd, ":");
619 if (ptr1 != NULL) {
620 n = (size_t)ptr3 - (size_t)ptr1;
621 n = n >= sizeof TempBuf ? sizeof TempBuf - 1 : n;
622 strncpy(TempBuf, ptr1, n);
623 TempBuf[n] = '\0';
624 if (strlen("urn") + strlen(TempBuf) >= sizeof(Evt->DeviceType))
625 return -1;
626 snprintf(Evt->DeviceType, sizeof(Evt->DeviceType), "urn%s",
627 TempBuf);
628 } else
629 return -1;
630 return 0;
631 }
632 if ((TempPtr = strstr(cmd, "uuid")) != NULL) {
633 if ((Ptr = strstr(cmd, "::")) != NULL) {
634 n = (size_t)Ptr - (size_t)TempPtr;
635 n = n >= sizeof Evt->UDN ? sizeof Evt->UDN - 1 : n;
636 strncpy(Evt->UDN, TempPtr, n);
637 Evt->UDN[n] = '\0';
638 } else {
639 memset(Evt->UDN, 0, sizeof(Evt->UDN));
640 strncpy(Evt->UDN, TempPtr, sizeof Evt->UDN - 1);
641 }
