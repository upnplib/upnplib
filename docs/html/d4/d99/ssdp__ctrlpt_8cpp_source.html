86int CreateClientRequestPacket(
88 char* RqstBuf,
90 size_t RqstBufSize,
92 int Mx,
94 char* SearchTarget,
96 int AddressFamily) {
97 int rc;
98 char TempBuf[COMMAND_LEN];
99 const char* command = "M-SEARCH * HTTP/1.1\r\n";
100 const char* man = "MAN: \"ssdp:discover\"\r\n";
101
102 memset(TempBuf, 0, sizeof(TempBuf));
103 if (RqstBufSize <= strlen(command))
104 return UPNP_E_INTERNAL_ERROR;
105 strcpy(RqstBuf, command);
106
107 switch (AddressFamily) {
108 case AF_INET:
109 rc = snprintf(TempBuf, sizeof(TempBuf), "HOST: %s:%d\r\n", SSDP_IP,
110 SSDP_PORT);
111 break;
112 case AF_INET6:
113 rc = snprintf(TempBuf, sizeof(TempBuf), "HOST: [%s]:%d\r\n",
114 SSDP_IPV6_LINKLOCAL, SSDP_PORT);
115 break;
116 default:
117 return UPNP_E_INVALID_ARGUMENT;
118 }
119 if (rc < 0 || (unsigned int)rc >= sizeof(TempBuf))
120 return UPNP_E_INTERNAL_ERROR;
121
122 if (RqstBufSize <= strlen(RqstBuf) + strlen(TempBuf))
123 return UPNP_E_BUFFER_TOO_SMALL;
124 strcat(RqstBuf, TempBuf);
125
126 if (RqstBufSize <= strlen(RqstBuf) + strlen(man))
127 return UPNP_E_BUFFER_TOO_SMALL;
128 strcat(RqstBuf, man);
129
130 if (Mx > 0) {
131 rc = snprintf(TempBuf, sizeof(TempBuf), "MX: %d\r\n", Mx);
132 if (rc < 0 || (unsigned int)rc >= sizeof(TempBuf))
133 return UPNP_E_INTERNAL_ERROR;
134 if (RqstBufSize <= strlen(RqstBuf) + strlen(TempBuf))
135 return UPNP_E_BUFFER_TOO_SMALL;
136 strcat(RqstBuf, TempBuf);
137 }
138
139 if (SearchTarget != NULL) {
140 rc = snprintf(TempBuf, sizeof(TempBuf), "ST: %s\r\n", SearchTarget);
141 if (rc < 0 || (unsigned int)rc >= sizeof(TempBuf))
142 return UPNP_E_INTERNAL_ERROR;
143 if (RqstBufSize <= strlen(RqstBuf) + strlen(TempBuf))
144 return UPNP_E_BUFFER_TOO_SMALL;
145 strcat(RqstBuf, TempBuf);
146 }
147 if (RqstBufSize <= strlen(RqstBuf) + strlen("\r\n"))
148 return UPNP_E_BUFFER_TOO_SMALL;
149 strcat(RqstBuf, "\r\n");
150
151 return UPNP_E_SUCCESS;
152}

236inline void searchExpired(
238 void* arg) {
239 int id = ((SsdpSearchExpArg*)arg)->timeoutEventId;
240 int handle = ((SsdpSearchExpArg*)arg)->handle;
241 struct Handle_Info* ctrlpt_info = NULL;
242
243 /* remove search Target from list and call client back */
244 ListNode* node = NULL;
245 SsdpSearchArg* item;
246 Upnp_FunPtr ctrlpt_callback;
247 void* cookie = NULL;
248 int found = 0;
249
250 HandleLock();
251
252 /* remove search target from search list */
253 if (GetHandleInfo(handle, &ctrlpt_info) != HND_CLIENT) {
254 free(arg);
255 HandleUnlock();
256 return;
257 }
258 ctrlpt_callback = ctrlpt_info->Callback;
259 node = ListHead(&ctrlpt_info->SsdpSearchList);
260 while (node != NULL) {
261 item = (SsdpSearchArg*)node->item;
262 if (item->timeoutEventId == id) {
263 free(item->searchTarget);
264 cookie = item->cookie;
265 found = 1;
266 item->searchTarget = NULL;
267 free(item);
268 ListDelNode(&ctrlpt_info->SsdpSearchList, node, 0);
269 break;
270 }
271 node = ListNext(&ctrlpt_info->SsdpSearchList, node);
272 }
273 HandleUnlock();
274
275 if (found)
276 ctrlpt_callback(UPNP_DISCOVERY_SEARCH_TIMEOUT, NULL, cookie);
277
278 free(arg);
279}

284void ssdp_handle_ctrlpt_msg(http_message_t* hmsg,
285 struct sockaddr_storage* dest_addr, int timeout) {
286 int handle;
287 int handle_start;
288 struct Handle_Info* ctrlpt_info = NULL;
289 memptr hdr_value;
290 /* byebye or alive */
291 int is_byebye;
292 UpnpDiscovery* param = UpnpDiscovery_new();
293 int expires;
294 int ret;
295 SsdpEvent event;
296 int nt_found;
297 int usn_found;
298 int st_found;
299 char save_char;
300 Upnp_EventType event_type;
301 Upnp_FunPtr ctrlpt_callback;
302 void* ctrlpt_cookie;
303 ListNode* node = NULL;
304 SsdpSearchArg* searchArg = NULL;
305 int matched = 0;
306 SSDPResultData* threadData = NULL;
307 ThreadPoolJob job;
308
309 /* we are assuming that there can be only one client supported at a time
310 */
311 HandleReadLock();
312
313 if (GetClientHandleInfo(&handle_start, &ctrlpt_info) != HND_CLIENT) {
314 HandleUnlock();
315 goto end_ssdp_handle_ctrlpt_msg;
316 }
317 HandleUnlock();
318 /* search timeout */
319 if (timeout) {
320 for (handle = handle_start; handle < NUM_HANDLE; handle++) {
321 HandleLock();
322
323 /* get client info */
324 if (GetHandleInfo(handle, &ctrlpt_info) != HND_CLIENT) {
325 HandleUnlock();
326 continue;
327 }
328 /* copy */
329 ctrlpt_callback = ctrlpt_info->Callback;
330 ctrlpt_cookie = ctrlpt_info->Cookie;
331 HandleUnlock();
332
333 ctrlpt_callback(UPNP_DISCOVERY_SEARCH_TIMEOUT, NULL, ctrlpt_cookie);
334 }
335 goto end_ssdp_handle_ctrlpt_msg;
336 }
337
338 UpnpDiscovery_set_ErrCode(param, UPNP_E_SUCCESS);
339 /* MAX-AGE, assume error */
340 expires = -1;
341 UpnpDiscovery_set_Expires(param, expires);
342 if (httpmsg_find_hdr(hmsg, HDR_CACHE_CONTROL, &hdr_value) != NULL) {
343 ret = matchstr(hdr_value.buf, hdr_value.length, "%imax-age = %d%0",
344 &expires);
345 UpnpDiscovery_set_Expires(param, expires);
346 if (ret != PARSE_OK)
347 goto end_ssdp_handle_ctrlpt_msg;
348 }
349 /* DATE */
350 if (httpmsg_find_hdr(hmsg, HDR_DATE, &hdr_value) != NULL) {
351 UpnpDiscovery_strcpy_Date(param, hdr_value.buf);
352 }
353 /* dest addr */
354 UpnpDiscovery_set_DestAddr(param, dest_addr);
355 /* EXT */
356 if (httpmsg_find_hdr(hmsg, HDR_EXT, &hdr_value) != NULL) {
357 UpnpDiscovery_strncpy_Ext(param, hdr_value.buf, hdr_value.length);
358 }
359 /* LOCATION */
360 if (httpmsg_find_hdr(hmsg, HDR_LOCATION, &hdr_value) != NULL) {
361 UpnpDiscovery_strncpy_Location(param, hdr_value.buf, hdr_value.length);
362 }
363 /* SERVER / USER-AGENT */
364 if (httpmsg_find_hdr(hmsg, HDR_SERVER, &hdr_value) != NULL ||
365 httpmsg_find_hdr(hmsg, HDR_USER_AGENT, &hdr_value) != NULL) {
366 UpnpDiscovery_strncpy_Os(param, hdr_value.buf, hdr_value.length);
367 }
368 /* clear everything */
369 event.UDN[0] = '\0';
370 event.DeviceType[0] = '\0';
371 event.ServiceType[0] = '\0';
372 nt_found = 0;
373 if (httpmsg_find_hdr(hmsg, HDR_NT, &hdr_value) != NULL) {
374 save_char = hdr_value.buf[hdr_value.length];
375 hdr_value.buf[hdr_value.length] = '\0';
376 nt_found = (ssdp_request_type(hdr_value.buf, &event) == 0);
377 hdr_value.buf[hdr_value.length] = save_char;
378 }
379 usn_found = 0;
380 if (httpmsg_find_hdr(hmsg, HDR_USN, &hdr_value) != NULL) {
381 save_char = hdr_value.buf[hdr_value.length];
382 hdr_value.buf[hdr_value.length] = '\0';
383 usn_found = (unique_service_name(hdr_value.buf, &event) == 0);
384 hdr_value.buf[hdr_value.length] = save_char;
385 }
386 if (nt_found || usn_found) {
387 UpnpDiscovery_strcpy_DeviceID(param, event.UDN);
388 UpnpDiscovery_strcpy_DeviceType(param, event.DeviceType);
389 UpnpDiscovery_strcpy_ServiceType(param, event.ServiceType);
390 }
391 /* ADVERT. OR BYEBYE */
392 if (hmsg->is_request) {
393 /* use NTS hdr to determine advert., or byebye */
394 if (httpmsg_find_hdr(hmsg, HDR_NTS, &hdr_value) == NULL) {
395 /* error; NTS header not found */
396 goto end_ssdp_handle_ctrlpt_msg;
397 }
398 if (memptr_cmp(&hdr_value, "ssdp:alive") == 0) {
399 is_byebye = 0;
400 } else if (memptr_cmp(&hdr_value, "ssdp:byebye") == 0) {
401 is_byebye = 1;
402 } else {
403 /* bad value */
404 goto end_ssdp_handle_ctrlpt_msg;
405 }
406 if (is_byebye) {
407 /* check device byebye */
408 if (!nt_found || !usn_found) {
409 /* bad byebye */
410 goto end_ssdp_handle_ctrlpt_msg;
411 }
412 event_type = UPNP_DISCOVERY_ADVERTISEMENT_BYEBYE;
413 } else {
414 /* check advertisement.
415 * Expires is valid if positive. This is for testing
416 * only. Expires should be greater than 1800 (30 mins)
417 */
418 if (!nt_found || !usn_found ||
419 UpnpString_get_Length(UpnpDiscovery_get_Location(param)) == 0 ||
420 UpnpDiscovery_get_Expires(param) <= 0) {
421 /* bad advertisement */
422 goto end_ssdp_handle_ctrlpt_msg;
423 }
424 event_type = UPNP_DISCOVERY_ADVERTISEMENT_ALIVE;
425 }
426 /* call callback */
427 for (handle = handle_start; handle < NUM_HANDLE; handle++) {
428 HandleLock();
429
430 /* get client info */
431 if (GetHandleInfo(handle, &ctrlpt_info) != HND_CLIENT) {
432 HandleUnlock();
433 continue;
434 }
435 /* copy */
436 ctrlpt_callback = ctrlpt_info->Callback;
437 ctrlpt_cookie = ctrlpt_info->Cookie;
438 HandleUnlock();
439
440 ctrlpt_callback(event_type, param, ctrlpt_cookie);
441 }
442 } else {
443 /* reply (to a SEARCH) */
444 /* only checking to see if there is a valid ST header */
445 st_found = 0;
446 if (httpmsg_find_hdr(hmsg, HDR_ST, &hdr_value) != NULL) {
447 save_char = hdr_value.buf[hdr_value.length];
448 hdr_value.buf[hdr_value.length] = '\0';
449 st_found = ssdp_request_type(hdr_value.buf, &event) == 0;
450 hdr_value.buf[hdr_value.length] = save_char;
451 }
452 if (hmsg->status_code != HTTP_OK ||
453 UpnpDiscovery_get_Expires(param) <= 0 ||
454 UpnpString_get_Length(UpnpDiscovery_get_Location(param)) == 0 ||
455 !usn_found || !st_found) {
456 /* bad reply */
457 goto end_ssdp_handle_ctrlpt_msg;
458 }
459 /* check each current search */
460 for (handle = handle_start; handle < NUM_HANDLE; handle++) {
461 HandleLock();
462
463 /* get client info */
464 if (GetHandleInfo(handle, &ctrlpt_info) != HND_CLIENT) {
465 HandleUnlock();
466 continue;
467 }
468 /* copy */
469 ctrlpt_callback = ctrlpt_info->Callback;
470 ctrlpt_cookie = ctrlpt_info->Cookie;
471
472 node = ListHead(&ctrlpt_info->SsdpSearchList);
473 /* temporary add null termination */
474 /*save_char = hdr_value.buf[ hdr_value.length ]; */
475 /*hdr_value.buf[ hdr_value.length ] = '\0'; */
476 while (node != NULL) {
477 searchArg = (SsdpSearchArg*)node->item;
478 /* check for match of ST header and search
479 * target */
480 switch (searchArg->requestType) {
481 case SSDP_ALL:
482 matched = 1;
483 break;
484 case SSDP_ROOTDEVICE:
485 matched = (event.RequestType == SSDP_ROOTDEVICE);
486 break;
487 case SSDP_DEVICEUDN:
488 matched = !strncmp(searchArg->searchTarget, hdr_value.buf,
489 hdr_value.length);
490 break;
491 case SSDP_DEVICETYPE: {
492 size_t m = std::min(hdr_value.length,
493 strlen(searchArg->searchTarget));
494 matched =
495 !strncmp(searchArg->searchTarget, hdr_value.buf, m);
496 break;
497 }
498 case SSDP_SERVICE: {
499 size_t m = std::min(hdr_value.length,
500 strlen(searchArg->searchTarget));
501 matched =
502 !strncmp(searchArg->searchTarget, hdr_value.buf, m);
503 break;
504 }
505 default:
506 matched = 0;
507 break;
508 }
509 if (matched) {
510 /* schedule call back */
511 threadData = SSDPResultData_new();
512 if (threadData != NULL) {
513 SSDPResultData_set_Param(threadData, param);
514 SSDPResultData_set_Cookie(threadData,
515 searchArg->cookie);
516 SSDPResultData_set_CtrlptCallback(threadData,
517 ctrlpt_callback);
518 memset(&job, 0, sizeof(job));
519
520 TPJobInit(&job, (start_routine)send_search_result,
521 threadData);
522 TPJobSetPriority(&job, MED_PRIORITY);
523 TPJobSetFreeFunction(&job, (free_routine)free);
524 if (ThreadPoolAdd(&gRecvThreadPool, &job, NULL) != 0) {
525 SSDPResultData_delete(threadData);
526 }
527 }
528 }
529 node = ListNext(&ctrlpt_info->SsdpSearchList, node);
530 }
531
532 HandleUnlock();
533 /*ctrlpt_callback( UPNP_DISCOVERY_SEARCH_RESULT, param,
534 * ctrlpt_cookie ); */
535 }
536 }
537
538end_ssdp_handle_ctrlpt_msg:
539 UpnpDiscovery_delete(param);
540}

543int SearchByTarget(int Hnd, int Mx, char* St, void* Cookie) {
544 char errorBuffer[ERROR_BUFFER_LEN];
545 int* id = NULL;
546 int ret = 0;
547 char ReqBufv4[BUFSIZE];
548#ifdef UPNP_ENABLE_IPV6
549 char ReqBufv6[BUFSIZE];
550 char ReqBufv6UlaGua[BUFSIZE];
551#endif
552 struct sockaddr_storage __ss_v4;
553#ifdef UPNP_ENABLE_IPV6
554 struct sockaddr_storage __ss_v6;
555#endif
556 struct sockaddr_in* destAddr4 = (struct sockaddr_in*)&__ss_v4;
557#ifdef UPNP_ENABLE_IPV6
558 struct sockaddr_in6* destAddr6 = (struct sockaddr_in6*)&__ss_v6;
559#endif
560 fd_set wrSet;
561 SsdpSearchArg* newArg = NULL;
562 SsdpSearchExpArg* expArg = NULL;
563 int timeTillRead = 0;
564 struct Handle_Info* ctrlpt_info = NULL;
565 enum SsdpSearchType requestType;
566 struct in_addr addrv4;
567 SOCKET max_fd = 0;
568 int retVal;
569
570 /*ThreadData *ThData; */
571 ThreadPoolJob job;
572
573 if (strlen(gIF_IPV4) > (size_t)0 &&
574 !inet_pton(AF_INET, gIF_IPV4, &addrv4)) {
575 return UPNP_E_INVALID_PARAM;
576 }
577
578 memset(&job, 0, sizeof(job));
579
580 requestType = ssdp_request_type1(St);
581 if (requestType == SSDP_SERROR)
582 return UPNP_E_INVALID_PARAM;
583 UpnpPrintf(UPNP_INFO, SSDP, __FILE__, __LINE__, "Inside SearchByTarget\n");
584 timeTillRead = Mx;
585 if (timeTillRead < MIN_SEARCH_TIME)
586 timeTillRead = MIN_SEARCH_TIME;
587 else if (timeTillRead > MAX_SEARCH_TIME)
588 timeTillRead = MAX_SEARCH_TIME;
589 retVal = CreateClientRequestPacket(ReqBufv4, sizeof(ReqBufv4), timeTillRead,
590 St, AF_INET);
591 if (retVal != UPNP_E_SUCCESS)
592 return retVal;
593#ifdef UPNP_ENABLE_IPV6
594 retVal = CreateClientRequestPacket(ReqBufv6, sizeof(ReqBufv6), timeTillRead,
595 St, AF_INET6);
596 if (retVal != UPNP_E_SUCCESS)
597 return retVal;
598 retVal = CreateClientRequestPacketUlaGua(
599 ReqBufv6UlaGua, sizeof(ReqBufv6UlaGua), timeTillRead, St, AF_INET6);
600 if (retVal != UPNP_E_SUCCESS)
601 return retVal;
602#endif
603
604 memset(&__ss_v4, 0, sizeof(__ss_v4));
605 destAddr4->sin_family = (sa_family_t)AF_INET;
606 inet_pton(AF_INET, SSDP_IP, &destAddr4->sin_addr);
607 destAddr4->sin_port = htons(SSDP_PORT);
608
609#ifdef UPNP_ENABLE_IPV6
610 memset(&__ss_v6, 0, sizeof(__ss_v6));
611 destAddr6->sin6_family = (sa_family_t)AF_INET6;
612 inet_pton(AF_INET6, SSDP_IPV6_SITELOCAL, &destAddr6->sin6_addr);
613 destAddr6->sin6_port = htons(SSDP_PORT);
614 destAddr6->sin6_scope_id = gIF_INDEX;
615#endif
616
617 /* add search criteria to list */
618 HandleLock();
619 if (GetHandleInfo(Hnd, &ctrlpt_info) != HND_CLIENT) {
620 HandleUnlock();
621 return UPNP_E_INTERNAL_ERROR;
622 }
623 newArg = (SsdpSearchArg*)malloc(sizeof(SsdpSearchArg));
624 newArg->searchTarget = strdup(St);
625 newArg->cookie = Cookie;
626 newArg->requestType = requestType;
627
628 expArg = (SsdpSearchExpArg*)malloc(sizeof(SsdpSearchExpArg));
629 expArg->handle = Hnd;
630 id = (int*)&(expArg->timeoutEventId);
631 TPJobInit(&job, (start_routine)searchExpired, expArg);
632 TPJobSetPriority(&job, MED_PRIORITY);
633 TPJobSetFreeFunction(&job, (free_routine)free);
634 /* Schedule a timeout event to remove search Arg */
635 TimerThreadSchedule(&gTimerThread, timeTillRead, REL_SEC, &job, SHORT_TERM,
636 id);
637 newArg->timeoutEventId = *id;
638 ListAddTail(&ctrlpt_info->SsdpSearchList, newArg);
639 HandleUnlock();
640 /* End of lock */
641
642 FD_ZERO(&wrSet);
643 if (gSsdpReqSocket4 != INVALID_SOCKET) {
644 umock::sys_socket_h.setsockopt(gSsdpReqSocket4, IPPROTO_IP,
645 IP_MULTICAST_IF, (char*)&addrv4,
646 sizeof(addrv4));
647 FD_SET(gSsdpReqSocket4, &wrSet);
648 max_fd = std::max(max_fd, gSsdpReqSocket4);
649 }
650#ifdef UPNP_ENABLE_IPV6
651 if (gSsdpReqSocket6 != INVALID_SOCKET) {
652 umock::sys_socket_h.setsockopt(gSsdpReqSocket6, IPPROTO_IPV6,
653 IPV6_MULTICAST_IF, (char*)&gIF_INDEX,
654 sizeof(gIF_INDEX));
655 FD_SET(gSsdpReqSocket6, &wrSet);
656 max_fd = std::max(max_fd, gSsdpReqSocket6);
657 }
658#endif
659 ret = umock::sys_socket_h.select((int)max_fd + 1, NULL, &wrSet, NULL, NULL);
660 if (ret == -1) {
661 strerror_r(errno, errorBuffer, ERROR_BUFFER_LEN);
662 UpnpPrintf(UPNP_INFO, SSDP, __FILE__, __LINE__,
663 "SSDP_LIB: Error in select(): %s\n", errorBuffer);
664 UpnpCloseSocket(gSsdpReqSocket4);
665#ifdef UPNP_ENABLE_IPV6
666 UpnpCloseSocket(gSsdpReqSocket6);
667#endif
668 return UPNP_E_INTERNAL_ERROR;
669 }
670#ifdef UPNP_ENABLE_IPV6
671 if (gSsdpReqSocket6 != INVALID_SOCKET &&
672 FD_ISSET(gSsdpReqSocket6, &wrSet)) {
673 int NumCopy = 0;
674
675 while (NumCopy < NUM_SSDP_COPY) {
676 UpnpPrintf(UPNP_INFO, SSDP, __FILE__, __LINE__,
677 ">>> SSDP SEND M-SEARCH >>>\n%s\n", ReqBufv6UlaGua);
678 sendto(gSsdpReqSocket6, ReqBufv6UlaGua,
