292static int WinsockInit() {
293 int retVal = UPNP_E_SUCCESS;
294#ifdef _WIN32
295 TRACE("Executing WinsockInit()")
296 WORD wVersionRequested;
297 WSADATA wsaData;
298 int err;
299
300 wVersionRequested = MAKEWORD(2, 2);
301 err = WSAStartup(wVersionRequested, &wsaData);
302 if (err != 0) {
303 /* Tell the user that we could not find a usable */
304 /* WinSock DLL. */
305 retVal = UPNP_E_INIT_FAILED;
306 goto exit_function;
307 }
308 /* Confirm that the WinSock DLL supports 2.2.
309 * Note that if the DLL supports versions greater
310 * than 2.2 in addition to 2.2, it will still return
311 * 2.2 in wVersion since that is the version we
312 * requested. */
313 if (LOBYTE(wsaData.wVersion) != 2 || HIBYTE(wsaData.wVersion) != 2) {
314 /* Tell the user that we could not find a usable
315 * WinSock DLL. */
316 WSACleanup();
317 retVal = UPNP_E_INIT_FAILED;
318 goto exit_function;
319 }
320 /* The WinSock DLL is acceptable. Proceed. */
321exit_function:
322#endif
323 return retVal;
324}

407static int UpnpInitPreamble() {
408 TRACE("Executing UpnpInitPreamble()")
409 int retVal = UPNP_E_SUCCESS;
410 int i;
411
412 retVal = WinsockInit();
413 if (retVal != UPNP_E_SUCCESS) {
414 return retVal;
415 }
416
417 /* needed by SSDP or other parts. */
418 srand((unsigned int)time(NULL));
419
420 /* Initialize debug output. */
421 retVal = UpnpInitLog();
422 if (retVal != UPNP_E_SUCCESS) {
423 /* UpnpInitLog does not return a valid UPNP_E_*. */
424 return UPNP_E_INIT_FAILED;
425 }
426
427 UPNPLIB_LOGINFO "MSG1058: Executing...\n";
428
429 /* Initialize SDK global mutexes. */
430 retVal = UpnpInitMutexes();
431 if (retVal != UPNP_E_SUCCESS) {
432 return retVal;
433 }
434
435#ifdef COMPA_HAVE_OPTION_SSDP
436 /* Create the NLS uuid. */
437 uuid_upnp nls_uuid;
438 uuid_create(&nls_uuid);
439 upnp_uuid_unpack(&nls_uuid, gUpnpSdkNLSuuid);
440#endif
441
442 /* Initializes the handle list. */
443 HandleLock();
444 for (i = 0; i < NUM_HANDLE; ++i) {
445 HandleTable[i] = NULL;
446 }
447 HandleUnlock();
448
449 /* Initialize SDK global thread pools. */
450 retVal = UpnpInitThreadPools();
451 if (retVal != UPNP_E_SUCCESS) {
452 return retVal;
453 }
454
455#ifdef COMPA_HAVE_DEVICE_SOAP
456 SetSoapCallback(soap_device_callback);
457#endif
458
459#ifdef COMPA_HAVE_DEVICE_GENA
460 SetGenaCallback(genaCallback);
461#endif
462
463 /* Initialize the SDK timer thread. */
464 retVal = TimerThreadInit(&gTimerThread, &gSendThreadPool);
465 if (retVal != UPNP_E_SUCCESS) {
466 UpnpFinish();
467
468 return retVal;
469 }
470
471 return UPNP_E_SUCCESS;
472}

514int UpnpInit2(const char* IfName, unsigned short DestPort) {
515 TRACE("Executing UpnpInit2()")
516 int retVal;
517
518 /* Initializes the ithread library */
519 ithread_initialize_library();
520
521 ithread_mutex_lock(&gSDKInitMutex);
522
523 /* Check if we're already initialized. */
524 if (UpnpSdkInit == 1) {
525 retVal = UPNP_E_INIT;
526 goto exit_function;
527 }
528
529 /* Set the UpnpSdkInit flag to 1 to indicate we're successfully
530 * initialized. */
531 UpnpSdkInit = 1;
532
533 /* Perform initialization preamble. */
534 retVal = UpnpInitPreamble();
535 if (retVal != UPNP_E_SUCCESS) {
536 goto exit_function;
537 }
538
539 UPNPLIB_LOGINFO "MSG1059: IfName=\""
540 << IfName << "\", DestPort=" << DestPort << ".\n";
541
542 /* Retrieve interface information (Addresses, index, etc). */
543 retVal = UpnpGetIfInfo(IfName);
544 if (retVal != UPNP_E_SUCCESS) {
545 goto exit_function;
546 }
547
548 /* Finish initializing the SDK. */
549 retVal = UpnpInitStartServers(DestPort);
550 if (retVal != UPNP_E_SUCCESS) {
551 goto exit_function;
552 }
553
554exit_function:
555 if (retVal != UPNP_E_SUCCESS && retVal != UPNP_E_INIT) {
556 UpnpFinish();
557 }
558 ithread_mutex_unlock(&gSDKInitMutex);
559
560 return retVal;
561}

607int UpnpFinish() {
608 TRACE("Executing UpnpFinish()")
609#ifdef COMPA_HAVE_DEVICE_SSDP
610 UpnpDevice_Handle device_handle;
611#endif
612#ifdef COMPA_HAVE_CTRLPT_SSDP
613 UpnpClient_Handle client_handle;
614#endif
615 [[maybe_unused]] struct Handle_Info* temp;
616#ifdef UPNP_ENABLE_OPEN_SSL
617 freeSslCtx();
618#endif
619 if (UpnpSdkInit != 1)
620 return UPNP_E_FINISH;
621 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__,
622 "Inside UpnpFinish: UpnpSdkInit is %d\n", UpnpSdkInit);
623 if (UpnpSdkInit == 1)
624 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__,
625 "UpnpFinish: UpnpSdkInit is ONE\n");
626 PrintThreadPoolStats(&gSendThreadPool, __FILE__, __LINE__,
627 "Send Thread Pool");
628 PrintThreadPoolStats(&gRecvThreadPool, __FILE__, __LINE__,
629 "Recv Thread Pool");
630 PrintThreadPoolStats(&gMiniServerThreadPool, __FILE__, __LINE__,
631 "MiniServer Thread Pool");
632#ifdef COMPA_HAVE_DEVICE_SSDP
633 while (GetDeviceHandleInfo(0, AF_INET, &device_handle, &temp) ==
634 HND_DEVICE) {
635 UpnpUnRegisterRootDevice(device_handle);
636 }
637 while (GetDeviceHandleInfo(0, AF_INET6, &device_handle, &temp) ==
638 HND_DEVICE) {
639 UpnpUnRegisterRootDevice(device_handle);
640 }
641#endif
642#ifdef COMPA_HAVE_CTRLPT_SSDP
643 while (HND_CLIENT == GetClientHandleInfo(&client_handle, &temp)) {
644 UpnpUnRegisterClient(client_handle);
645 }
646#endif
647 TimerThreadShutdown(&gTimerThread);
648#ifdef COMPA_HAVE_MINISERVER
649 StopMiniServer();
650#endif
651#ifdef COMPA_HAVE_WEBSERVER
652 web_server_destroy();
653#endif
654 ThreadPoolShutdown(&gMiniServerThreadPool);
655 PrintThreadPoolStats(&gMiniServerThreadPool, __FILE__, __LINE__,
656 "MiniServer Thread Pool");
657 ThreadPoolShutdown(&gRecvThreadPool);
658 PrintThreadPoolStats(&gSendThreadPool, __FILE__, __LINE__,
659 "Send Thread Pool");
660 ThreadPoolShutdown(&gSendThreadPool);
661 PrintThreadPoolStats(&gRecvThreadPool, __FILE__, __LINE__,
662 "Recv Thread Pool");
663#ifdef COMPA_HAVE_CTRLPT_SSDP
664 ithread_mutex_destroy(&GlobalClientSubscribeMutex);
665#endif
666 ithread_rwlock_destroy(&GlobalHndRWLock);
667 ithread_mutex_destroy(&gUUIDMutex);
668 /* remove all virtual dirs */
669 UpnpRemoveAllVirtualDirs();
670 UpnpSdkInit = 0;
671 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__,
672 "Exiting UpnpFinish: UpnpSdkInit is :%d:\n", UpnpSdkInit);
673 UpnpCloseLog();
674 /* Clean-up ithread library resources */
675 ithread_cleanup_library();
676
677 return UPNP_E_SUCCESS;
678}
