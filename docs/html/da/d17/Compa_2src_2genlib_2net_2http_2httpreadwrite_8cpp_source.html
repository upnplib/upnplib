106int Check_Connect_And_Wait_Connection(
107 const SOCKET a_sock,
108 const int connect_res
110) {
111 TRACE("Executing Check_Connect_And_Wait_Connection()")
112
113 if (connect_res == 0)
114 return 0;
115#ifdef _WIN32
116 if (umock::winsock2_h.WSAGetLastError() != WSAEWOULDBLOCK)
117 return -1;
118#else
119 if (errno != EINPROGRESS)
120 return -1;
121#endif
122
123 fd_set fdSet;
124 FD_ZERO(&fdSet);
125 FD_SET(a_sock, &fdSet);
126 timeval tmvTimeout = {DEFAULT_TCP_CONNECT_TIMEOUT, 0};
127 int result{SOCKET_ERROR};
128
129 result =
130 umock::sys_socket_h.select(a_sock + 1, NULL, &fdSet, NULL, &tmvTimeout);
131 switch (result) {
132 case SOCKET_ERROR:
133 return -1;
134 case 0:
135 /* timeout */
136 return -1;
137 }
138 int valopt{};
139 socklen_t len{sizeof(valopt)};
140 if (umock::sys_socket_h.getsockopt(a_sock, SOL_SOCKET, SO_ERROR,
141 (void*)&valopt, &len) < 0)
142 /* failed to read delayed error */
143 return -1;
144 if (valopt)
145 /* delayed error = valopt */
146 // TODO: Return more detailed error codes, e.g. valopt == 111:
147 // ECONNREFUSED "Connection refused" if there is a remote host but no
148 // server service listening.
149 return -1;
150
151 return 0;
152}

173int get_hoststr(
174 const char* url_str,
175 const char**
176 hoststr,
178 size_t* hostlen
180) {
181 TRACE("Executing get_hoststr()")
182 const char* start;
183 const char* finish;
184 int ret_code = UPNP_E_INVALID_URL;
185
186 // strlen() must be at least 2 char otherwise compiler complains boundary
187 // error with next strstr() statement.
188 if (strlen(url_str) < 2)
189 goto end_function;
190
191 start = strstr(url_str, "//");
192 if (!start)
193 goto end_function;
194
195 start += 2;
196 finish = strchr(start, '/');
197 if (finish) {
198 *hostlen = static_cast<size_t>(finish - start);
199 } else {
200 *hostlen = strlen(start);
201 }
202 *hoststr = start;
203
204 ret_code = UPNP_E_SUCCESS;
205
206end_function:
207 return ret_code;
208}

248int MakeGenericMessage(http_method_t method, const char* url_str,
249 membuffer* request, uri_type* url, int contentLength,
250 const char* contentType, const UpnpString* headers) {
251 int ret_code = 0;
252 size_t hostlen{};
253 const char* hoststr;
254
255 UpnpPrintf(UPNP_INFO, HTTP, __FILE__, __LINE__, "URL: %s method: %d\n",
256 url_str, method);
257 ret_code = http_FixStrUrl(url_str, strlen(url_str), url);
258 if (ret_code != UPNP_E_SUCCESS)
259 return ret_code;
260 /* make msg */
261 membuffer_init(request);
262 ret_code = http_MakeMessage(request, 1, 1, "Q", method, url->pathquery.buff,
263 url->pathquery.size);
264 /* add request headers if specified, otherwise use default headers */
265 if (ret_code == 0) {
266 if (headers) {
267 ret_code = http_MakeMessage(request, 1, 1, "s",
268 UpnpString_get_String(headers));
269 } else {
270 ret_code = get_hoststr(url_str, &hoststr, &hostlen);
271 if (ret_code != UPNP_E_SUCCESS)
272 return ret_code;
273 UpnpPrintf(UPNP_INFO, HTTP, __FILE__, __LINE__,
274 "HOSTNAME : %s Length : %" PRIzu "\n", hoststr, hostlen);
275 ret_code = http_MakeMessage(request, 1, 1,
276 "s"
277 "bcDCU",
278 "HOST: ", hoststr, hostlen);
279 }
280 }
281
282 /* add the content-type header */
283 if (ret_code == 0 && contentType) {
284 ret_code = http_MakeMessage(request, 1, 1, "T", contentType);
285 }
286 /* add content-length header. */
287 if (ret_code == 0) {
288 if (contentLength >= 0)
289 ret_code =
290 http_MakeMessage(request, 1, 1, "Nc", (off_t)contentLength);
291 else if (contentLength == UPNP_USING_CHUNKED)
292 ret_code = http_MakeMessage(request, 1, 1, "Kc");
293 else if (contentLength == UPNP_UNTIL_CLOSE)
294 ret_code = http_MakeMessage(request, 1, 1, "c");
295 else
296 ret_code = UPNP_E_INVALID_PARAM;
297 }
298 if (ret_code != 0) {
299 UpnpPrintf(UPNP_INFO, HTTP, __FILE__, __LINE__,
300 "HTTP Makemessage failed\n");
301 membuffer_destroy(request);
302 return ret_code;
303 }
304 UpnpPrintf(UPNP_INFO, HTTP, __FILE__, __LINE__,
305 "HTTP Buffer:\n%s\n"
306 "----------END--------\n",
307 request->buf);
308
309 return UPNP_E_SUCCESS;
310}

335int ReadResponseLineAndHeaders(
337 SOCKINFO* info,
339 http_parser_t* parser,
341 int* timeout_secs,
343 int* http_error_code) {
344 parse_status_t status;
345 int num_read;
346 char buf[2 * 1024];
347 int done = 0;
348 int ret_code = 0;
349
350 /*read response line */
351 status = parser_parse_responseline(parser);
352 switch (status) {
353 case PARSE_OK:
354 done = 1;
355 break;
356 case PARSE_INCOMPLETE:
357 done = 0;
358 break;
359 default:
360 /*error */
361 return status;
362 }
363 while (!done) {
364 num_read = sock_read(info, buf, sizeof(buf), timeout_secs);
365 if (num_read > 0) {
366 /* append data to buffer */
367 ret_code =
368 membuffer_append(&parser->msg.msg, buf, (size_t)num_read);
369 if (ret_code != 0) {
370 /* set failure status */
371 parser->http_error_code = HTTP_INTERNAL_SERVER_ERROR;
372 return PARSE_FAILURE;
373 }
374 status = parser_parse_responseline(parser);
375 switch (status) {
376 case PARSE_OK:
377 done = 1;
378 break;
379 case PARSE_INCOMPLETE:
380 done = 0;
381 break;
382 default:
383 /*error */
384 return status;
385 }
386 } else if (num_read == 0) {
387 /* partial msg */
388 *http_error_code = HTTP_BAD_REQUEST; /* or response */
389 return UPNP_E_BAD_HTTPMSG;
390 } else {
391 *http_error_code = parser->http_error_code;
392 return num_read;
393 }
394 }
395 status = parser_parse_headers(parser);
396 if ((status == (parse_status_t)PARSE_OK) &&
397 (parser->position == (parser_pos_t)POS_ENTITY))
398 done = 1;
399 else if (status == (parse_status_t)PARSE_INCOMPLETE)
400 done = 0;
401 else
402 /*error */
403 return status;
404 /*read headers */
405 while (!done) {
406 num_read = sock_read(info, buf, sizeof(buf), timeout_secs);
407 if (num_read > 0) {
408 /* append data to buffer */
409 ret_code =
410 membuffer_append(&parser->msg.msg, buf, (size_t)num_read);
411 if (ret_code != 0) {
412 /* set failure status */
413 parser->http_error_code = HTTP_INTERNAL_SERVER_ERROR;
414 return PARSE_FAILURE;
415 }
416 status = parser_parse_headers(parser);
417 if (status == (parse_status_t)PARSE_OK &&
418 parser->position == (parser_pos_t)POS_ENTITY)
419 done = 1;
420 else if (status == (parse_status_t)PARSE_INCOMPLETE)
421 done = 0;
422 else
423 /*error */
424 return status;
425 } else if (num_read == 0) {
426 /* partial msg */
427 *http_error_code = HTTP_BAD_REQUEST; /* or response */
428 return UPNP_E_BAD_HTTPMSG;
429 } else {
430 *http_error_code = parser->http_error_code;
431 return num_read;
432 }
433 }
434
435 return PARSE_OK;
436}

504static int private_connect(
505 const SOCKET sockfd,
506 const sockaddr* const
507 serv_addr,
508 const socklen_t addrlen
509) {
510 TRACE("Executing private_connect(), blocking " +
511 std::string(unblock_tcp_connections ? "false" : "true"))
512
513 if (unblock_tcp_connections) {
514 // This is never used due to SDK specification. It is only presevered
515 // for historical reasons.
516
517 int ret{SOCKET_ERROR};
518 // returns 0 if successful, else SOCKET_ERROR.
519 ret = umock::pupnp_sock.sock_make_no_blocking(sockfd);
520 if (ret == 0) {
521 // ret is needed for Check_Connect_And_Wait_Connection(),
522 // returns 0 if successful, else -1.
523 ret = umock::sys_socket_h.connect(sockfd, serv_addr, addrlen);
524 // returns 0 if successful, else -1.
525 ret = Check_Connect_And_Wait_Connection(sockfd, ret);
526
527 // Always make_blocking() to revert make_no_blocking() above.
528 // returns 0 if successful, else SOCKET_ERROR.
529 ret = ret | umock::pupnp_sock.sock_make_blocking(sockfd);
530 }
531
532 return ret == 0 ? 0 : SOCKET_ERROR;
533
534 } else { // tcp_connection is blocking
535
536 upnplib::CSocketErr serrObj;
537 if (umock::sys_socket_h.connect(sockfd, serv_addr, addrlen) != 0) {
538 serrObj.catch_error();
539 UPNPLIB_LOGERR "MSG1020: failed to connect() socket("
540 << sockfd << "): " << serrObj.error_str() << "\n";
541
542 return SOCKET_ERROR;
543 }
544 return 0;
545 }

560int http_FixUrl(uri_type* url, uri_type* fixed_url) {
561 const char* temp_path = "/";
562
563 *fixed_url = *url;
564#ifdef UPNP_ENABLE_OPEN_SSL
565 if (token_string_casecmp(&fixed_url->scheme, "http") != 0 &&
566 token_string_casecmp(&fixed_url->scheme, "https") != 0) {
567 return UPNP_E_INVALID_URL;
568 }
569#else
570 if (token_string_casecmp(&fixed_url->scheme, "http") != 0) {
571 return UPNP_E_INVALID_URL;
572 }
573#endif
574 if (fixed_url->hostport.text.size == (size_t)0) {
575 return UPNP_E_INVALID_URL;
576 }
577 /* set pathquery to "/" if it is empty */
578 if (fixed_url->pathquery.size == (size_t)0) {
579 fixed_url->pathquery.buff = temp_path;
580 fixed_url->pathquery.size = (size_t)1;
581 }
582
583 return UPNP_E_SUCCESS;
584}

630int http_RecvMessage(SOCKINFO* info, http_parser_t* parser,
631 http_method_t request_method, int* timeout_secs,
632 int* http_error_code) {
633 TRACE("Executing http_RecvMessage()")
634 int ret{UPNP_E_SUCCESS};
635 int line{};
636 parse_status_t status{};
637 int num_read{};
638 int ok_on_close{};
639 char* buf{nullptr};
640 size_t buf_len{1024};
641
642 *http_error_code = HTTP_INTERNAL_SERVER_ERROR;
643 buf = (char*)malloc(buf_len);
644 if (!buf) {
645 ret = UPNP_E_OUTOF_MEMORY;
646 goto ExitFunction;
647 }
648 if (request_method == (http_method_t)HTTPMETHOD_UNKNOWN) {
649 parser_request_init(parser);
650 } else {
651 parser_response_init(parser, request_method);
652 }
653
654 while (1) {
655 /* Double the bet if needed */
656 /* We should have already exited the loop if the value was <= 0
657 * so the cast is safe */
658 if ((size_t)num_read >= buf_len) {
659 free(buf);
660 buf_len = 2 * buf_len;
661 buf = (char*)malloc(buf_len);
662 if (!buf) {
663 ret = UPNP_E_OUTOF_MEMORY;
664 goto ExitFunction;
665 }
666 }
667 num_read = sock_read(info, buf, buf_len, timeout_secs);
668 if (num_read > 0) {
669 /* got data */
670 status = parser_append(parser, buf, (size_t)num_read);
671 switch (status) {
672 case PARSE_SUCCESS:
673 UPNPLIB_LOGINFO "MSG1031: <<< (RECVD) <<<\n"
674 << parser->msg.msg.buf << "UPnPlib -----------------\n";
675 print_http_headers(&parser->msg);
676 if (g_maxContentLength > (size_t)0 &&
677 parser->content_length > (unsigned int)g_maxContentLength) {
678 *http_error_code = HTTP_REQ_ENTITY_TOO_LARGE;
679 line = __LINE__;
680 ret = UPNP_E_OUTOF_BOUNDS;
681 goto ExitFunction;
682 }
683 line = __LINE__;
684 ret = 0;
685 goto ExitFunction;
686 case PARSE_FAILURE:
687 case PARSE_NO_MATCH:
688 *http_error_code = parser->http_error_code;
689 line = __LINE__;
690 ret = UPNP_E_BAD_HTTPMSG;
691 goto ExitFunction;
692 case PARSE_INCOMPLETE_ENTITY:
693 /* read until close */
694 ok_on_close = 1;
695 break;
696 case PARSE_CONTINUE_1:
697 /* Web post request. */
698 line = __LINE__;
699 ret = PARSE_SUCCESS;
700 goto ExitFunction;
701 default:
702 break;
703 }
704 } else if (num_read == 0) {
705 if (ok_on_close) {
706 UPNPLIB_LOGINFO "MSG1047: <<< (RECVD) <<<\n"
707 << parser->msg.msg.buf << "\n-----------------\n";
708 print_http_headers(&parser->msg);
709 line = __LINE__;
710 ret = UPNP_E_SUCCESS;
711 goto ExitFunction;
712 } else {
713 /* partial msg */
714 *http_error_code = HTTP_BAD_REQUEST; /* or response */
715 line = __LINE__;
716 ret = UPNP_E_BAD_HTTPMSG;
717 goto ExitFunction;
718 }
719 } else {
720 *http_error_code = parser->http_error_code;
721 line = __LINE__;
722 ret = num_read;
723 goto ExitFunction;
724 }
725 }
726
727ExitFunction:
728 free(buf);
729 if (ret != UPNP_E_SUCCESS) {
730 UPNPLIB_LOGERR << "MSG1048: " << ret << " on line " << line
731 << ", http_error_code = " << *http_error_code << ".\n";
732 }
733
734 return ret;
735}

737int http_SendMessage(SOCKINFO* info, int* TimeOut, const char* fmt, ...) {
738 TRACE("Executing http_SendMessage()")
739#ifdef COMPA_HAVE_WEBSERVER
740 FILE* Fp{nullptr};
741 SendInstruction* Instr{nullptr};
742 char* filename{nullptr};
743 char* file_buf{nullptr};
744 char* ChunkBuf{nullptr};
745 /* 10 byte allocated for chunk header. */
746 char Chunk_Header[CHUNK_HEADER_SIZE];
747 size_t num_read;
748 off_t amount_to_be_read{};
749 size_t Data_Buf_Size{WEB_SERVER_BUF_SIZE};
750#endif /* COMPA_HAVE_WEBSERVER */
751 va_list argp;
752 char* buf{nullptr};
753 char c;
754 int nw;
755 int RetVal = UPNP_E_SUCCESS;
756 size_t buf_length;
757 size_t num_written;
758
759#ifdef COMPA_HAVE_WEBSERVER
760 int I_fmt_processed = 0;
761 memset(Chunk_Header, 0, sizeof(Chunk_Header));
762#endif
763 va_start(argp, fmt);
764 while ((c = *fmt++) != '\0') {
765#ifdef COMPA_HAVE_WEBSERVER
766 if (c == 'I' && !I_fmt_processed) {
767 I_fmt_processed = 1;
768 Instr = va_arg(argp, struct SendInstruction*);
769 if (Instr->ReadSendSize >= 0) {
770 amount_to_be_read = Instr->ReadSendSize;
771 } else {
772 amount_to_be_read = (off_t)Data_Buf_Size;
773 }

775 Data_Buf_Size = (size_t)amount_to_be_read;

777 (size_t)(Data_Buf_Size + CHUNK_HEADER_SIZE + CHUNK_TAIL_SIZE));
778 if (!ChunkBuf) {
779 RetVal = UPNP_E_OUTOF_MEMORY;
780 goto ExitFunction;
781 }
782 file_buf = ChunkBuf + CHUNK_HEADER_SIZE;
783 } else if (c == 'f') {
784 /* file name */
785 filename = va_arg(argp, char*);
786 if (Instr && Instr->IsVirtualFile)
787 Fp = (FILE*)(virtualDirCallback.open)(
788 filename, UPNP_READ, Instr->Cookie, Instr->RequestCookie);
789 else
790#ifdef _WIN32
791 // returns error code direct
792 if (umock::stdio_h.fopen_s(&Fp, filename, "rb") != 0)
793 Fp = nullptr;
794#else
795 Fp = umock::stdio_h.fopen(filename, "rb");
796#endif
797 if (Fp == nullptr) {
798 RetVal = UPNP_E_FILE_READ_ERROR;
799 goto ExitFunction;
800 }
801 if (Instr && Instr->IsRangeActive && Instr->IsVirtualFile) {
802 if (virtualDirCallback.seek(Fp, Instr->RangeOffset, SEEK_CUR,
803 Instr->Cookie,
804 Instr->RequestCookie) != 0) {
805 RetVal = UPNP_E_FILE_READ_ERROR;
806 goto Cleanup_File;
807 }
808 } else if (Instr && Instr->IsRangeActive) {
809 if (fseeko(Fp, Instr->RangeOffset, SEEK_CUR) != 0) {
810 RetVal = UPNP_E_FILE_READ_ERROR;
811 goto Cleanup_File;
812 }
813 }
814 while (amount_to_be_read) {
815 if (Instr) {
816 int nr;
817 size_t n = amount_to_be_read >= (off_t)Data_Buf_Size
818 ? Data_Buf_Size
819 : (size_t)amount_to_be_read;
820 if (Instr->IsVirtualFile) {
821 nr = virtualDirCallback.read(Fp, file_buf, n,
822 Instr->Cookie,
823 Instr->RequestCookie);
824 num_read = (size_t)nr;
825 } else {
826 num_read =
827 umock::stdio_h.fread(file_buf, (size_t)1, n, Fp);
828 TRACE("Read " + std::to_string(num_read) +
829 " bytes from input file = \"" +
830 std::string(file_buf) + "\"")
831 }
832 amount_to_be_read -= (off_t)num_read;
833 if (Instr->ReadSendSize < 0) {
834 /* read until close */
835 amount_to_be_read = (off_t)Data_Buf_Size;
836 }
837 } else {
838 num_read = umock::stdio_h.fread(file_buf, (size_t)1,
839 Data_Buf_Size, Fp);
840 }
841 if (num_read == (size_t)0) {
842 /* EOF so no more to send. */
843 if (Instr && Instr->IsChunkActive) {
844 const char* str = "0\r\n\r\n";
845 nw = sock_write(info, str, strlen(str), TimeOut);
846 } else {
847 RetVal = UPNP_E_FILE_READ_ERROR;
848 }
849 goto Cleanup_File;
850 }
851 /* Create chunk for the current buffer. */
852 if (Instr && Instr->IsChunkActive) {
853 int rc;
854 /* Copy CRLF at the end of the chunk */
855 memcpy(file_buf + num_read, "\r\n", (size_t)2);
856 /* Hex length for the chunk size. */
857 memset(Chunk_Header, 0, sizeof(Chunk_Header));
858 rc = snprintf(Chunk_Header, sizeof(Chunk_Header),
859 "%" PRIzx "\r\n", (unsigned long)num_read);
860 if (rc < 0 || (unsigned int)rc >= sizeof(Chunk_Header)) {
861 RetVal = UPNP_E_INTERNAL_ERROR;
862 goto Cleanup_File;
863 }
864 /* Copy the chunk size header */
865 memcpy(file_buf - strlen(Chunk_Header), Chunk_Header,
866 strlen(Chunk_Header));
867 /* on the top of the buffer. */
868 /*file_buf[num_read+strlen(Chunk_Header)]
869 * = NULL; */
870 /*upnpprintf("Sending
871 * %s\n",file_buf-strlen(Chunk_Header));*/
872 nw = sock_write(info, file_buf - strlen(Chunk_Header),
873 num_read + strlen(Chunk_Header) + (size_t)2,
874 TimeOut);
875 num_written = static_cast<size_t>(nw);
876 if (nw <= 0 || num_written != num_read +
877 strlen(Chunk_Header) +
878 (size_t)2)
879 /* Send error nothing we can do.
880 */
881 goto Cleanup_File;
882 } else {
883 /* write data */
884 nw = sock_write(info, file_buf, num_read, TimeOut);
885 UPNPLIB_LOGINFO "MSG1104: >>> (SENT) >>>\n"
886 << std::string(file_buf, static_cast<size_t>(nw))
887 << "UPnPlib ------------\n";
888 /* Send error nothing we can do */
889 num_written = (size_t)nw;
890 if (nw <= 0 || num_written != num_read) {
891 goto Cleanup_File;
892 }
893 }
894 } /* while */
895 Cleanup_File:
896 if (Instr && Instr->IsVirtualFile) {
897 virtualDirCallback.close(Fp, Instr->Cookie,
898 Instr->RequestCookie);
899 } else {
900 umock::stdio_h.fclose(Fp);
901 }
902 goto ExitFunction;
903 } else
904#endif /* COMPA_HAVE_WEBSERVER */
905 if (c == 'b') {
906 /* Message to send is given in a memory buffer */
907 buf = va_arg(argp, char*);
908 buf_length = va_arg(argp, size_t);
909 if (buf_length > (size_t)0) {
910 nw = sock_write(info, buf, buf_length, TimeOut);
911 num_written = (size_t)nw;
912
913 if (upnplib::g_dbug) {
914 upnplib::SSockaddr saObj;
915 memcpy(&saObj.ss, &info->foreign_sockaddr,
916 saObj.sizeof_ss());
917 UPNPLIB_LOGINFO "MSG1105: >>> (SENT) >>> to \""
918 << saObj.netaddrp() << "\"\n"
919 << std::string(buf, buf_length)
920 << "UPnPlib buf_length=" << buf_length
921 << ", num_written=" << num_written
922 << ".\nUPnPlib ------------\n";
923 }
924 if (nw < 0) {
925 RetVal = nw;
926 goto ExitFunction;
927 }
928 if (num_written != buf_length) {
929 RetVal = UPNP_E_SOCKET_WRITE;
930 goto ExitFunction;
931 }
932 }
933 }
934 }
935
936ExitFunction:
937 va_end(argp);
938#ifdef COMPA_HAVE_WEBSERVER
939 free(ChunkBuf);
940#endif
941 return RetVal;
942}

1335int http_ReadHttpResponse(void* Handle, char* buf, size_t* size, int timeout) {
1336 http_connection_handle_t* handle = (http_connection_handle_t*)Handle;
1337 parse_status_t status;
1338 int num_read;
1339 int ok_on_close = 0;
1340 char tempbuf[2 * 1024];
1341 int ret_code = 0;
1342
1343 if (!handle || !size || (*size > 0 && !buf)) {
1344 if (size)
1345 *size = 0;
1346 return UPNP_E_INVALID_PARAM;
1347 }
1348 /* first parse what has already been gotten */
1349 if (handle->response.position != POS_COMPLETE)
1350 status = parser_parse_entity(&handle->response);
1351 else
1352 status = PARSE_SUCCESS;
1353 if (status == PARSE_INCOMPLETE_ENTITY)
1354 /* read until close */
1355 ok_on_close = 1;
1356 else if ((status != PARSE_SUCCESS) && (status != PARSE_CONTINUE_1) &&
1357 (status != PARSE_INCOMPLETE)) {
1358 /*error */
1359 *size = 0;
1360 return UPNP_E_BAD_RESPONSE;
1361 }
1362 /* read more if necessary entity */
1363 while (handle->response.msg.amount_discarded + *size >
1364 handle->response.msg.entity.length &&
1365 !handle->cancel && handle->response.position != POS_COMPLETE) {
1366 num_read =
1367 sock_read(&handle->sock_info, tempbuf, sizeof(tempbuf), &timeout);
1368 if (num_read > 0) {
1369 /* append data to buffer */
1370 ret_code = membuffer_append(&handle->response.msg.msg, tempbuf,
1371 (size_t)num_read);
1372 if (ret_code != 0) {
1373 /* set failure status */
1374 handle->response.http_error_code = HTTP_INTERNAL_SERVER_ERROR;
1375 *size = 0;
1376 return PARSE_FAILURE;
1377 }
1378 status = parser_parse_entity(&handle->response);
1379 if (status == PARSE_INCOMPLETE_ENTITY) {
1380 /* read until close */
1381 ok_on_close = 1;
1382 } else if ((status != PARSE_SUCCESS) &&
1383 (status != PARSE_CONTINUE_1) &&
1384 (status != PARSE_INCOMPLETE)) {
1385 /*error */
1386 *size = 0;
1387 return UPNP_E_BAD_RESPONSE;
1388 }
1389 } else if (num_read == 0) {
1390 if (ok_on_close) {
1391 UpnpPrintf(UPNP_INFO, HTTP, __FILE__, __LINE__,
1392 "<<< (RECVD) "
1393 "<<<\n%s\n-----------------\n",
1394 handle->response.msg.msg.buf);
1395 handle->response.position = POS_COMPLETE;
1396 } else {
1397 /* partial msg */
1398 *size = 0;
1399 handle->response.http_error_code =
1400 HTTP_BAD_REQUEST; /* or response */
1401 return UPNP_E_BAD_HTTPMSG;
1402 }
1403 } else {
1404 *size = 0;
1405 return num_read;
1406 }
1407 }
1408 if (handle->cancel) {
1409 return UPNP_E_CANCELED;
1410 }
1411 /* truncate size to fall within available data */
1412 if (handle->response.msg.amount_discarded + *size >
1413 handle->response.msg.entity.length)
1414 *size = handle->response.msg.entity.length -
1415 handle->response.msg.amount_discarded;
1416 /* copy data to user buffer. delete copied data */
1417 if (*size > 0) {
1418 memcpy(buf,
1419 &handle->response.msg.msg
1420 .buf[handle->response.entity_start_position],

1481 off_t bignum;
1482 size_t length;

1488 const char* end_str;

1517 pos = UpnpListNext(head, pos)) {

1522 goto error_handler;

1527 }
1528 } else if (c == 's') {

1531 assert(s);
1532 UpnpPrintf(UPNP_ALL, HTTP, __FILE__, __LINE__,

1535 goto error_handler;
1536 } else if (c == 'K') {

1541 } else if (c == 'G') {

1551 } else if (c == 'b') {
1552 /* mem buffer */

1555 "Adding a char Buffer starting with: %c\n", (int)s[0]);
1556 assert(s);
1557 length = (size_t)va_arg(argp, size_t);
1558 if (membuffer_append(buf, s, length))
1559 goto error_handler;
1560 } else if (c == 'c') {
1561 /* crlf */
1562 if (membuffer_append(buf, "\r\n", (size_t)2))
1563 goto error_handler;
1564 } else if (c == 'd') {
1565 /* integer */
1566 num = (size_t)va_arg(argp, int);
1567 rc = snprintf(tempbuf, sizeof(tempbuf), "%" PRIzu,
1568 (unsigned long)num);

1571 goto error_handler;
1572 } else if (c == 'h') {
1573 /* off_t */
1574 bignum = (off_t)va_arg(argp, off_t);
1575 rc =
1576 snprintf(tempbuf, sizeof(tempbuf), "%" PRId64, (int64_t)bignum);
1577 if (rc < 0 || (unsigned int)rc >= sizeof(tempbuf) ||
1578 membuffer_append(buf, tempbuf, strlen(tempbuf)))
1579 goto error_handler;
1580 } else if (c == 't' || c == 'D') {
1581 /* date */
1582 if (c == 'D') {
1583 /* header */
1584 start_str = "DATE: ";
1585 end_str = "\r\n";
1586 curr_time = umock::sysinfo.time(NULL);
1587 loc_time = &curr_time;
1588 } else {
1589 /* date value only */
1590 start_str = end_str = "";
1591 loc_time = (time_t*)va_arg(argp, time_t*);
1592 }
1593 assert(loc_time);
1594 date = http_gmtime_r(loc_time, &date_storage);
1595 if (date == NULL)
1596 goto error_handler;

1599 &weekday_str[date->tm_wday * 4], date->tm_mday,
1600 &month_str[date->tm_mon * 4], date->tm_year + 1900,

1607 // if Accept-Language header is not empty

1632 if (http_MakeMessage(buf, http_major_version, http_minor_version,

1639 } else if (c == 'S' || c == 'U') {

1658 assert(status_code > 0);

1696 } else if (c == 'q') {

1699 uri_ptr = (uri_type*)va_arg(argp, uri_type*);
1700 assert(uri_ptr);

1704 }

1724
1725error_handler:

1732 va_end(argp);

1734}

1757 membuffer request;

1761 int errCode = UPNP_E_SUCCESS;
1762 struct SendInstruction rangeBuf;
1763 int rc = 0;
1764
1765 membuffer_init(&request);
1766
1767 do {
1768 /* Checking Input parameters */
1769 if (!url_str || !Handle || !contentType || !httpStatus) {
1770 errCode = UPNP_E_INVALID_PARAM;
1771 break;
1772 }
1773 /* Initialize output parameters */
1774 *httpStatus = 0;
1775 *Handle = handle;
1776 *contentType = NULL;
1777 *contentLength = 0;
1778 if (lowRange > highRange) {
1779 errCode = UPNP_E_INTERNAL_ERROR;
1780 break;
1781 }
1782 memset(&rangeBuf, 0, sizeof(rangeBuf));
1783 rc = snprintf(rangeBuf.RangeHeader, sizeof(rangeBuf.RangeHeader),
1784 "Range: bytes=%d-%d\r\n", lowRange, highRange);
1785 if (rc < 0 || (unsigned int)rc >= sizeof(rangeBuf.RangeHeader))
1786 break;
1787 membuffer_init(&request);
1788 errCode = MakeGetMessageEx(url_str, &request, &url, &rangeBuf);
1789 if (errCode != UPNP_E_SUCCESS)
1790 break;
1791 handle =
1792 (http_connection_handle_t*)malloc(sizeof(http_connection_handle_t));
1793 if (!handle) {
1794 errCode = UPNP_E_OUTOF_MEMORY;
1795 break;
1796 }
1797 memset(handle, 0, sizeof(*handle));
1798 parser_response_init(&handle->response, HTTPMETHOD_GET);
1799 tcp_connection = umock::sys_socket_h.socket(
1800 (int)url.hostport.IPaddress.ss_family, SOCK_STREAM, 0);
1801 if (tcp_connection == INVALID_SOCKET) {
1802 errCode = UPNP_E_SOCKET_ERROR;
1803 free(handle);
1804 break;
1805 }
1806 if (sock_init(&handle->sock_info, tcp_connection) != UPNP_E_SUCCESS) {
1807 sock_destroy(&handle->sock_info, SD_BOTH);
1808 errCode = UPNP_E_SOCKET_ERROR;
1809 free(handle);
1810 break;
1811 }
1812 sockaddr_len = url.hostport.IPaddress.ss_family == AF_INET6
1813 ? sizeof(struct sockaddr_in6)
1814 : sizeof(struct sockaddr_in);
1815 errCode = umock::pupnp_httprw.private_connect(
1816 handle->sock_info.socket,
1817 (struct sockaddr*)&(url.hostport.IPaddress),
1818 (socklen_t)sockaddr_len);
1819 if (errCode == -1) {
1820 sock_destroy(&handle->sock_info, SD_BOTH);
1821 errCode = UPNP_E_SOCKET_CONNECT;
1822 free(handle);
1823 break;
1824 }
1825 /* send request */
1826 errCode = http_SendMessage(&handle->sock_info, &timeout, "b",
1827 request.buf, request.length);
1828 if (errCode != UPNP_E_SUCCESS) {
1829 sock_destroy(&handle->sock_info, SD_BOTH);
1830 free(handle);
1831 break;
1832 }
1833 if (ReadResponseLineAndHeaders(&handle->sock_info, &handle->response,
1834 &timeout,
1835 &http_error_code) != (int)PARSE_OK) {
1836 errCode = UPNP_E_BAD_RESPONSE;
1837 free(handle);
1838 break;
1839 }

1841 if (status != (parse_status_t)PARSE_CONTINUE_1 &&
1842 status != (parse_status_t)PARSE_SUCCESS) {
1843 errCode = UPNP_E_BAD_RESPONSE;
1844 free(handle);
1845 break;
1846 }
1847 *httpStatus = handle->response.msg.status_code;
1848 errCode = UPNP_E_SUCCESS;
1849
1850 if (!httpmsg_find_hdr(&handle->response.msg, HDR_CONTENT_TYPE, &ctype))
1851 /* no content-type */

1853 else
1854 *contentType = ctype.buf;

1861 else if (handle->response.ent_position == ENTREAD_UNTIL_CLOSE)
1862 *contentLength = UPNP_UNTIL_CLOSE;
1863 *Handle = handle;
1864 } while (0);
1865
1866 membuffer_destroy(&request);
1867
1868 return errCode;
1869}
