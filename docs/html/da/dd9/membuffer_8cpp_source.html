59char* str_alloc(const char* str, size_t str_len) {

61

63 if (s == NULL) {
64 return NULL; /* no mem */

66

74 int cmp;
75
76 cmp = strncmp(m->buf, s, m->length);

118int membuffer_set_size(membuffer* m, size_t new_length) {
119 size_t diff;
120 size_t alloc_len;
121 char* temp_buf;
122
123 if (new_length >= m->length) { /* increase length */
124 /* need more mem? */
125 if (new_length <= m->capacity) {
126 return 0; /* have enough mem; done */
127 }
128
129 diff = new_length - m->length;
130 alloc_len = std::max(m->size_inc, diff) + m->capacity;
131 } else { /* decrease length */
132
133 assert(new_length <= m->length);
134
135 /* if diff is 0..m->size_inc, don't free */
136 if ((m->capacity - new_length) <= m->size_inc) {
137 return 0;
138 }
139
140 alloc_len = new_length + m->size_inc;
141 }
142
143 assert(alloc_len >= new_length);
144
145 temp_buf = (char*)umock::stdlib_h.realloc(
146 m->buf, alloc_len + (size_t)1); /*LEAK_FIX_MK */
147
148 /*temp_buf = Realloc( m->buf,m->length, alloc_len + 1 );LEAK_FIX_MK */
149
150 if (temp_buf == NULL) {
151 /* try smaller size */
152 alloc_len = new_length;
153 temp_buf = (char*)umock::stdlib_h.realloc(
154 m->buf, alloc_len + (size_t)1); /*LEAK_FIX_MK */
155 /*temp_buf = Realloc( m->buf,m->length, alloc_len + 1
156 * );LEAK_FIX_MK */
157
158 if (temp_buf == NULL) {
159 return UPNP_E_OUTOF_MEMORY;
160 }
161 }
162 /* save */
163 m->buf = temp_buf;
164 m->capacity = alloc_len;
165 return 0;
166}

188int membuffer_assign(membuffer* m, const void* buf, size_t buf_len) {
189 int return_code;
190
191 assert(m != NULL);
192
193 /* set value to null */
194 if (buf == NULL) {
195 membuffer_destroy(m);
196 return 0;
197 }
198 /* alloc mem */
199 return_code = membuffer_set_size(m, buf_len);
200 if (return_code != 0)
201 return return_code;
202 /* copy */
203 if (buf_len) {
204 memcpy(m->buf, buf, buf_len);
205 m->buf[buf_len] = 0; /* null-terminate */
206 }
207 m->length = buf_len;
208
209 return 0;
210}

226int membuffer_insert(membuffer* m, const void* buf, size_t buf_len,
227 size_t index) {
228 int return_code;
229
230 assert(m != NULL);
231
232 if (index > m->length)
233 return UPNP_E_OUTOF_BOUNDS;
234 if (!buf || !buf_len) {
235 return 0;
236 }
237 /* alloc mem */
238 return_code = membuffer_set_size(m, m->length + buf_len);
239 if (return_code) {
240 return return_code;
241 }
242 /* insert data */
243 /* move data to right of insertion point */
244 memmove(m->buf + index + buf_len, m->buf + index, m->length - index);
245 memcpy(m->buf + index, buf, buf_len);
246 m->length += buf_len;
247 /* null-terminate */
248 m->buf[m->length] = 0;
249
250 return 0;
251}

253void membuffer_delete(membuffer* m, size_t index, size_t num_bytes) {
254 int return_value;
255 size_t new_length;
256 size_t copy_len;
257
258 assert(m != NULL);
259
260 if (!m || !m->length)
261 return;
262 /* shrink count if it goes beyond buffer */
263 if (index + num_bytes > m->length) {
264 num_bytes = m->length - index;
265 /* every thing at and after index purged */
266 copy_len = (size_t)0;
267 } else {
268 /* calc num bytes after deleted string */
269 copy_len = m->length - (index + num_bytes);
270 }
271 memmove(m->buf + index, m->buf + index + num_bytes, copy_len);
272 new_length = m->length - num_bytes;
273 /* trim buffer */
274 return_value = membuffer_set_size(m, new_length);
275 /* shrinking should always work */
276 assert(return_value == 0);
277 if (return_value != 0)
278 return;
279
280 /* don't modify until buffer is set */
281 m->length = new_length;
282 m->buf[new_length] = 0;
283}
