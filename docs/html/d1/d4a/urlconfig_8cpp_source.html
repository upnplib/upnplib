65inline int addrToString( //
66 const sockaddr* addr,
68 char* ipaddr_port,
70 size_t ipaddr_port_size)
71{
72 char buf_ntop[INET6_ADDRSTRLEN];
73 int rc = 0;
74
75 if (addr->sa_family == AF_INET) {
76 struct sockaddr_in* sa4 = (struct sockaddr_in*)addr;
77 inet_ntop(AF_INET, &sa4->sin_addr, buf_ntop, sizeof(buf_ntop));
78 rc = snprintf(ipaddr_port, ipaddr_port_size, "%s:%d", buf_ntop,
79 (int)ntohs(sa4->sin_port));
80 } else if (addr->sa_family == AF_INET6) {
81 struct sockaddr_in6* sa6 = (struct sockaddr_in6*)addr;
82 inet_ntop(AF_INET6, &sa6->sin6_addr, buf_ntop, sizeof(buf_ntop));
83 rc = snprintf(ipaddr_port, ipaddr_port_size, "[%s]:%d", buf_ntop,
84 (int)ntohs(sa6->sin6_port));
85 }
87 // if (rc < 0 || (unsigned int)rc >= ipaddr_port_size)
88 if (rc < 0 || (unsigned int)rc > ipaddr_port_size)
89 return UPNP_E_BUFFER_TOO_SMALL;
90 return UPNP_E_SUCCESS;
91}

102inline int calc_alias( //
103 const char* alias,
104 const char* rootPath,
105 char** newAlias
107) {
108 const char* aliasPtr;
109 size_t root_len;
110 const char* temp_str;
111 size_t new_alias_len;
112 char* alias_temp;
113
114 assert(rootPath);
115 assert(alias);
116
117 /* add / suffix, if missing */
118 root_len = strlen(rootPath);
119 if (root_len == 0 || rootPath[root_len - 1] != '/')
120 temp_str = "/";
121 else
122 temp_str = ""; /* suffix already present */
123 /* discard / prefix, if present */
124 if (alias[0] == '/')
125 aliasPtr = alias + 1;
126 else
127 aliasPtr = alias;
128 new_alias_len = root_len + strlen(temp_str) + strlen(aliasPtr) + (size_t)1;
129 alias_temp = (char*)malloc(new_alias_len);
130 if (alias_temp == nullptr)
131 return UPNP_E_OUTOF_MEMORY;
132 memset(alias_temp, 0, new_alias_len);
133 snprintf(alias_temp, new_alias_len, "%s%s%s", rootPath, temp_str, aliasPtr);
134
135 *newAlias = alias_temp;
136 return UPNP_E_SUCCESS;
137}

155

158

163 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__, "desc url: %s\n", descURL);
164
165 return UPNP_E_SUCCESS;

181inline int config_description_doc(
183 IXML_Document* doc,
185 const char* ip_str,
188 char** root_path_str) {
189 IXML_NodeList* baseList;
190 IXML_Element* element = NULL;
191 IXML_Node* textNode = NULL;
192 IXML_Node* rootNode = NULL;
193 IXML_Node* urlbase_node = NULL;
194 const char* urlBaseStr = "URLBase";
195 const DOMString domStr = NULL;
196 uri_type uri;
197 int err_code;
198 int len;
199 membuffer url_str;
200 membuffer root_path;
201
202 membuffer_init(&url_str);
203 membuffer_init(&root_path);
204 err_code = UPNP_E_OUTOF_MEMORY; /* default error */
205 baseList = ixmlDocument_getElementsByTagName(doc, urlBaseStr);
206 if (baseList == NULL) {
207 /* urlbase not found -- create new one */
208 element = ixmlDocument_createElement(doc, urlBaseStr);
209 if (element == NULL) {
210 goto error_handler;
211 }
212 if (membuffer_append_str(&url_str, "http://") != 0 ||
213 membuffer_append_str(&url_str, ip_str) != 0 ||
214 membuffer_append_str(&url_str, "/") != 0 ||
215 membuffer_append_str(&root_path, "/") != 0) {
216 goto error_handler;
217 }
218 rootNode = ixmlNode_getFirstChild((IXML_Node*)doc);
219 if (rootNode == NULL) {
220 err_code = UPNP_E_INVALID_DESC;
221 goto error_handler;
222 }
223 err_code = ixmlNode_appendChild(rootNode, (IXML_Node*)element);
224 if (err_code != IXML_SUCCESS) {
225 err_code = UPNP_E_INVALID_DESC;
226 goto error_handler;
227 }
228 textNode = ixmlDocument_createTextNode(doc, (char*)url_str.buf);
229 if (textNode == NULL) {
230 goto error_handler;
231 }
232 err_code = ixmlNode_appendChild((IXML_Node*)element, textNode);
233 if (err_code != IXML_SUCCESS) {
234 err_code = UPNP_E_INTERNAL_ERROR;
235 goto error_handler;
236 }
237 } else {
238 /* urlbase found */
239 urlbase_node = ixmlNodeList_item(baseList, 0lu);
240 assert(urlbase_node != NULL);
241 textNode = ixmlNode_getFirstChild(urlbase_node);
242 if (textNode == NULL) {
243 err_code = UPNP_E_INVALID_DESC;
244 goto error_handler;
245 }
246 domStr = ixmlNode_getNodeValue(textNode);
247 if (domStr == NULL) {
248 err_code = UPNP_E_INVALID_URL;
249 goto error_handler;
250 }
251 len = parse_uri(domStr, strlen(domStr), &uri);
252 if (len < 0 || uri.type != ABSOLUTE) {
253 err_code = UPNP_E_INVALID_URL;
254 goto error_handler;
255 }
256 if (membuffer_assign(&url_str, uri.scheme.buff, uri.scheme.size) != 0 ||
257 membuffer_append_str(&url_str, "://") != 0 ||
258 membuffer_append_str(&url_str, ip_str) != 0) {
259 goto error_handler;
260 }
261 /* add leading '/' if missing from relative path */
262 if ((uri.pathquery.size > 0 && uri.pathquery.buff[0] != '/') ||
263 (uri.pathquery.size == 0)) {
264 if (membuffer_append_str(&url_str, "/") != 0 ||
265 membuffer_append_str(&root_path, "/") != 0) {
266 goto error_handler;
267 }
268 }
269 if (membuffer_append(&url_str, uri.pathquery.buff,
270 uri.pathquery.size) != 0 ||
271 membuffer_append(&root_path, uri.pathquery.buff,
272 uri.pathquery.size) != 0) {
273 goto error_handler;
274 }
275 /* add trailing '/' if missing */
276 if (url_str.buf[url_str.length - 1] != '/') {
277 if (membuffer_append(&url_str, "/", (size_t)1) != 0) {
278 goto error_handler;
279 }
280 }
281 err_code = ixmlNode_setNodeValue(textNode, url_str.buf);
282 if (err_code != IXML_SUCCESS) {
283 err_code = UPNP_E_OUTOF_MEMORY;
284 goto error_handler;
285 }
286 }
287 *root_path_str = membuffer_detach(&root_path); /* return path */
288 err_code = UPNP_E_SUCCESS;
289
290error_handler:
291 if (err_code != UPNP_E_SUCCESS) {
292 ixmlElement_free(element);
293 }
294 ixmlNodeList_free(baseList);
295 membuffer_destroy(&root_path);
296 membuffer_destroy(&url_str);
297
298 return err_code;
299}

304int configure_urlbase(IXML_Document* doc, const sockaddr* serverAddr,
305 const char* alias, time_t last_modified,
306 char docURL[LINE_SIZE]) {
307 char* root_path = NULL;
308 char* new_alias = NULL;
309 char* xml_str = NULL;
310 int err_code;
311 char ipaddr_port[LINE_SIZE];
312
313 /* get IP address and port */
314 err_code = addrToString(serverAddr, ipaddr_port, sizeof(ipaddr_port));
315 if (err_code != UPNP_E_SUCCESS) {
316 goto error_handler;
317 }
318
319 /* config url-base in 'doc' */
320 err_code = config_description_doc(doc, ipaddr_port, &root_path);
321 if (err_code != UPNP_E_SUCCESS) {
322 goto error_handler;
323 }
324 /* calc alias */
325 err_code = calc_alias(alias, root_path, &new_alias);
326 if (err_code != UPNP_E_SUCCESS) {
327 goto error_handler;
328 }
329 /* calc full url for desc doc */
330 err_code = calc_descURL(ipaddr_port, new_alias, docURL);
331 if (err_code != UPNP_E_SUCCESS) {
332 goto error_handler;
333 }
334 /* xml doc to str */
335 xml_str = ixmlPrintDocument(doc);
336 if (xml_str == NULL) {
337 goto error_handler;
338 }
339
340 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__, "desc url: %s\n", docURL);
341 UpnpPrintf(UPNP_INFO, API, __FILE__, __LINE__, "doc = %s\n", xml_str);
342 /* store in web server */
343 err_code = web_server_set_alias(new_alias, xml_str, strlen(xml_str),
344 last_modified);
345
346error_handler:
347 free(root_path);
348 free(new_alias);
349
350 if (err_code != UPNP_E_SUCCESS) {
351 ixmlFreeDOMString(xml_str);
352 }
353 return err_code;
354}
