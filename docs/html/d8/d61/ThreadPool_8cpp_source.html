74long DiffMillis(timeval* time1, timeval* time2) {
75 double temp = 0.0;
76
77 temp = time1->tv_sec - time2->tv_sec;
78 /* convert to milliseconds */
79 temp *= 1000.0;
80
81 /* convert microseconds to milliseconds and add to temp */
82 /* implicit flooring of unsigned long data type */
83 temp += (time1->tv_usec - time2->tv_usec) / 1000.0;
84
85 return temp;
86}

157void CalcWaitTime(
159 ThreadPool* tp,
161 ThreadPriority p,
163 ThreadPoolJob* job) {
164 struct timeval now;
165 long diff;
166
167 assert(tp != NULL);
168 assert(job != NULL);
169
170 gettimeofday(&now, NULL);
171 diff = DiffMillis(&now, &job->requestTime);
172 switch (p) {
173 case LOW_PRIORITY:
174 StatsAccountLQ(tp, diff);
175 break;
176 case MED_PRIORITY:
177 StatsAccountMQ(tp, diff);
178 break;
179 case HIGH_PRIORITY:
180 StatsAccountHQ(tp, diff);
181 break;
182 default:
183 assert(0);
184 }

239int SetPolicyType(
241 [[maybe_unused]] PolicyType in) {
242 int retVal = 0;
243#ifdef __CYGWIN__
245 (void)in;
246 retVal = 0;
247#elif defined(__APPLE__) || defined(__NetBSD__)
248 (void)in;
249 setpriority(PRIO_PROCESS, 0, 0);
250 retVal = 0;
251#elif defined(__PTW32_DLLPORT)
252 retVal = sched_setscheduler(0, in);
253#elif defined(_POSIX_PRIORITY_SCHEDULING) && _POSIX_PRIORITY_SCHEDULING > 0
254 struct sched_param current;
255 int sched_result;
256
257 memset(&current, 0, sizeof(current));
258 sched_getparam(0, &current);
259 current.sched_priority = sched_get_priority_min(DEFAULT_POLICY);
260 sched_result = sched_setscheduler(0, in, &current);
261 retVal = (sched_result != -1 || errno == EPERM) ? 0 : errno;
262#else
263 retVal = 0;
264#endif

275int SetPriority(
277 ThreadPriority priority) {
278#if defined(_POSIX_PRIORITY_SCHEDULING) && _POSIX_PRIORITY_SCHEDULING > 0
279 int retVal = 0;
280 int currentPolicy;
281 int minPriority = 0;
282 int maxPriority = 0;
283 int actPriority = 0;
284 int midPriority = 0;
285 struct sched_param newPriority;
286 int sched_result;
287
288 pthread_getschedparam(ithread_self(), &currentPolicy, &newPriority);
289 minPriority = sched_get_priority_min(currentPolicy);
290 maxPriority = sched_get_priority_max(currentPolicy);
291 midPriority = (maxPriority - minPriority) / 2;
292 switch (priority) {
293 case LOW_PRIORITY:
294 actPriority = minPriority;
295 break;
296 case MED_PRIORITY:
297 actPriority = midPriority;
298 break;
299 case HIGH_PRIORITY:
300 actPriority = maxPriority;
301 break;
302 default:
303 retVal = EINVAL;
304 goto exit_function;
305 };
306
307 newPriority.sched_priority = actPriority;
308
309 sched_result =
310 pthread_setschedparam(ithread_self(), currentPolicy, &newPriority);
311 retVal = (sched_result == 0 || errno == EPERM) ? 0 : sched_result;
312exit_function:
313 return retVal;
314#else
315 (void)priority;
316 return 0;
317#endif
318}

326void BumpPriority(
328 ThreadPool* tp) {
329 int done = 0;
330 struct timeval now;
331 long diffTime = 0;
332 ThreadPoolJob* tempJob = NULL;
333
334 gettimeofday(&now, NULL);
335 while (!done) {
336 if (tp->medJobQ.size) {
337 tempJob = (ThreadPoolJob*)tp->medJobQ.head.next->item;
338 diffTime = DiffMillis(&now, &tempJob->requestTime);
339 if (diffTime >= tp->attr.starvationTime) {
340 /* If job has waited longer than the starvation time, bump
341 * priority (add to higher priority Q) */
342 StatsAccountMQ(tp, diffTime);
343 ListDelNode(&tp->medJobQ, tp->medJobQ.head.next, 0);
344 ListAddTail(&tp->highJobQ, tempJob);
345 continue;
346 }
347 }
348 if (tp->lowJobQ.size) {
349 tempJob = (ThreadPoolJob*)tp->lowJobQ.head.next->item;
350 diffTime = DiffMillis(&now, &tempJob->requestTime);
351 if (diffTime >= tp->attr.maxIdleTime) {
352 /* If job has waited longer than the starvation time, bump
353 * priority (add to higher priority Q) */
354 StatsAccountLQ(tp, diffTime);
355 ListDelNode(&tp->lowJobQ, tp->lowJobQ.head.next, 0);
356 ListAddTail(&tp->medJobQ, tempJob);
357 continue;
358 }
359 }
360 done = 1;
361 }
362}

373 timeval now;
374 int sec = relMillis / 1000;
375 int milliSeconds = relMillis % 1000;
376
377 gettimeofday(&now, NULL);

424void* WorkerThread(
426 void* arg) {
427 time_t start = 0;
428
429 ThreadPoolJob* job = NULL;
430 ListNode* head = NULL;
431
432 timespec timeout;
433 int retCode = 0;
434 int persistent = -1;
435 ThreadPool* tp = (ThreadPool*)arg;
436
437 ithread_initialize_thread();
438
439 /* Increment total thread count */
440 ithread_mutex_lock(&tp->mutex);
441 tp->totalThreads++;
442 tp->pendingWorkerThreadStart = 0;
443 ithread_cond_broadcast(&tp->start_and_shutdown);
444 ithread_mutex_unlock(&tp->mutex);
445
446 SetSeed();
447 StatsTime(&start);
448 while (1) {
449 ithread_mutex_lock(&tp->mutex);
450 if (job) {
451 tp->busyThreads--;
452 FreeThreadPoolJob(tp, job);
453 job = NULL;
454 }
455 retCode = 0;
456 tp->stats.idleThreads++;
457 tp->stats.totalWorkTime += (double)StatsTime(NULL) - (double)start;
458 StatsTime(&start);
459 if (persistent == 0) {
460 tp->stats.workerThreads--;
461 } else if (persistent == 1) {
462 /* Persistent thread becomes a regular thread */
463 tp->persistentThreads--;
464 }
465
466 /* Check for a job or shutdown */
467 while (tp->lowJobQ.size == 0 && tp->medJobQ.size == 0 &&
468 tp->highJobQ.size == 0 && !tp->persistentJob && !tp->shutdown) {
469 /* If wait timed out and we currently have more than the
470 * min threads, or if we have more than the max threads
471 * (only possible if the attributes have been reset)
472 * let this thread die. */
473 if ((retCode == ETIMEDOUT &&
474 tp->totalThreads > tp->attr.minThreads) ||
475 (tp->attr.maxThreads != -1 &&
476 tp->totalThreads > tp->attr.maxThreads)) {
477 tp->stats.idleThreads--;
478 goto exit_function;
479 }
480 SetRelTimeout(&timeout, tp->attr.maxIdleTime);
481
482 /* wait for a job up to the specified max time */
483 retCode =
484 ithread_cond_timedwait(&tp->condition, &tp->mutex, &timeout);
485 }
486 tp->stats.idleThreads--;
487 /* idle time */
488 tp->stats.totalIdleTime += (double)StatsTime(NULL) - (double)start;
489 /* work time */
490 StatsTime(&start);
491 /* bump priority of starved jobs */
492 BumpPriority(tp);
493 /* if shutdown then stop */
494 if (tp->shutdown) {
495 goto exit_function;
496 } else {
497 /* Pick up persistent job if available */
498 if (tp->persistentJob) {
499 job = tp->persistentJob;
500 tp->persistentJob = NULL;
501 tp->persistentThreads++;
502 persistent = 1;
503 ithread_cond_broadcast(&tp->start_and_shutdown);
504 } else {
505 tp->stats.workerThreads++;
506 persistent = 0;
507 /* Pick the highest priority job */
508 if (tp->highJobQ.size > 0) {
509 head = ListHead(&tp->highJobQ);
510 if (head == NULL) {
511 tp->stats.workerThreads--;
512 goto exit_function;
513 }
514 job = (ThreadPoolJob*)head->item;
515 CalcWaitTime(tp, HIGH_PRIORITY, job);
516 ListDelNode(&tp->highJobQ, head, 0);
517 } else if (tp->medJobQ.size > 0) {
518 head = ListHead(&tp->medJobQ);
519 if (head == NULL) {
520 tp->stats.workerThreads--;
521 goto exit_function;
522 }
523 job = (ThreadPoolJob*)head->item;
524 CalcWaitTime(tp, MED_PRIORITY, job);
525 ListDelNode(&tp->medJobQ, head, 0);
526 } else if (tp->lowJobQ.size > 0) {
527 head = ListHead(&tp->lowJobQ);
528 if (head == NULL) {
529 tp->stats.workerThreads--;
530 goto exit_function;
531 }
532 job = (ThreadPoolJob*)head->item;
533 CalcWaitTime(tp, LOW_PRIORITY, job);
534 ListDelNode(&tp->lowJobQ, head, 0);
535 } else {
536 /* Should never get here */
537 tp->stats.workerThreads--;
538 goto exit_function;
539 }
540 }
541 }
542
543 tp->busyThreads++;
544 ithread_mutex_unlock(&tp->mutex);
545
546 /* In the future can log info */
547 if (SetPriority(job->priority) != 0) {
548 } else {
549 }
550 /* run the job */
551 job->func(job->arg);
552 /* return to Normal */
553 SetPriority(DEFAULT_PRIORITY);
554 }
555
556exit_function:
557 tp->totalThreads--;
558 ithread_cond_broadcast(&tp->start_and_shutdown);
559 ithread_mutex_unlock(&tp->mutex);
560 ithread_cleanup_thread();
561
562 return NULL;
563}

604int CreateWorker(
606 ThreadPool* tp) {
607 ithread_t temp;
608 int rc = 0;
609 ithread_attr_t attr;
610
611 /* if a new worker is the process of starting, wait until it fully
612 * starts */
613 while (tp->pendingWorkerThreadStart) {
614 ithread_cond_wait(&tp->start_and_shutdown, &tp->mutex);
615 }
616
617 if (tp->attr.maxThreads != INFINITE_THREADS &&
618 tp->totalThreads + 1 > tp->attr.maxThreads) {
619 return EMAXTHREADS;
620 }
621 ithread_attr_init(&attr);
622 ithread_attr_setstacksize(&attr, tp->attr.stackSize);
623 ithread_attr_setdetachstate(&attr, ITHREAD_CREATE_DETACHED);
624 rc = ithread_create(&temp, &attr, WorkerThread, tp);
625 ithread_attr_destroy(&attr);
626 if (rc == 0) {
627 tp->pendingWorkerThreadStart = 1;
628 /* wait until the new worker thread starts */
629 while (tp->pendingWorkerThreadStart) {
630 ithread_cond_wait(&tp->start_and_shutdown, &tp->mutex);
631 }
632 }
633 if (tp->stats.maxThreads < tp->totalThreads) {
634 tp->stats.maxThreads = tp->totalThreads;
635 }
636
637 return rc;
638}

649void AddWorker(
651 ThreadPool* tp) {
652 long jobs = 0;
653 int threads = 0;
654
655 jobs = tp->highJobQ.size + tp->lowJobQ.size + tp->medJobQ.size;
656 threads = tp->totalThreads - tp->persistentThreads;
657 while (threads == 0 || (jobs / threads) >= tp->attr.jobsPerThread ||
658 (tp->totalThreads == tp->busyThreads)) {
659 if (CreateWorker(tp) != 0) {
660 return;
661 }
662 threads++;
663 }
664}

670int ThreadPoolInit(ThreadPool* tp, ThreadPoolAttr* attr) {
671 TRACE2("Executing ThreadPoolInit() for ThreadPool ", tp)
672 int retCode = 0;
673 int i = 0;
674
675 if (!tp) {
676 return EINVAL;
677 }
678
679 retCode += ithread_mutex_init(&tp->mutex, NULL);
680 retCode += ithread_mutex_lock(&tp->mutex);
681
682 retCode += ithread_cond_init(&tp->condition, NULL);
683 retCode += ithread_cond_init(&tp->start_and_shutdown, NULL);
684 if (retCode) {
685 ithread_mutex_unlock(&tp->mutex);
686 ithread_mutex_destroy(&tp->mutex);
687 ithread_cond_destroy(&tp->condition);
688 ithread_cond_destroy(&tp->start_and_shutdown);
689 return EAGAIN;
690 }
691 if (attr) {
692 tp->attr = *attr;
693 } else {
694 TPAttrInit(&tp->attr);
695 }
696 if (SetPolicyType(tp->attr.schedPolicy) != 0) {
697 ithread_mutex_unlock(&tp->mutex);
698 ithread_mutex_destroy(&tp->mutex);
699 ithread_cond_destroy(&tp->condition);
700 ithread_cond_destroy(&tp->start_and_shutdown);
701
702 return INVALID_POLICY;
703 }
704 retCode +=
705 FreeListInit(&tp->jobFreeList, sizeof(ThreadPoolJob), JOBFREELISTSIZE);
706 StatsInit(&tp->stats);
707 retCode += ListInit(&tp->highJobQ, CmpThreadPoolJob, NULL);
708 retCode += ListInit(&tp->medJobQ, CmpThreadPoolJob, NULL);
709 retCode += ListInit(&tp->lowJobQ, CmpThreadPoolJob, NULL);
710 if (retCode) {
711 retCode = EAGAIN;
712 } else {
713 tp->persistentJob = NULL;
714 tp->lastJobId = 0;
715 tp->shutdown = 0;
716 tp->totalThreads = 0;
717 tp->busyThreads = 0;
718 tp->persistentThreads = 0;
719 tp->pendingWorkerThreadStart = 0;
720 for (i = 0; i < tp->attr.minThreads; ++i) {
721 retCode = CreateWorker(tp);
722 if (retCode) {
723 break;
724 }
725 }
726 }
727
728 ithread_mutex_unlock(&tp->mutex);
729
730 if (retCode) {
731 /* clean up if the min threads could not be created */
732 ThreadPoolShutdown(tp);
733 }
734
735 return retCode;
736}

738int ThreadPoolAddPersistent(ThreadPool* tp, ThreadPoolJob* job, int* jobId) {
739 int ret = 0;
740 int tempId = -1;
741 ThreadPoolJob* temp = NULL;
742
743 if (!tp || !job) {
744 return EINVAL;
745 }
746 if (!jobId) {
747 jobId = &tempId;
748 }
749 *jobId = INVALID_JOB_ID;
750
751 ithread_mutex_lock(&tp->mutex);
752
753 /* Create A worker if less than max threads running */
754 if (tp->totalThreads < tp->attr.maxThreads) {
755 CreateWorker(tp);
756 } else {
757 /* if there is more than one worker thread
758 * available then schedule job, otherwise fail */
759 if (tp->totalThreads - tp->persistentThreads - 1 == 0) {
760 ret = EMAXTHREADS;
761 goto exit_function;
762 }
763 }
764 temp = CreateThreadPoolJob(job, tp->lastJobId, tp);
765 if (!temp) {
766 ret = EOUTOFMEM;
767 goto exit_function;
768 }
769 tp->persistentJob = temp;
770
771 /* Notify a waiting thread */
772 ithread_cond_signal(&tp->condition);
773
774 /* wait until long job has been picked up */
775 while (tp->persistentJob)
776 ithread_cond_wait(&tp->start_and_shutdown, &tp->mutex);
777 *jobId = tp->lastJobId++;
778
779exit_function:
780 ithread_mutex_unlock(&tp->mutex);
781
782 return ret;
783}

785int ThreadPoolAdd(ThreadPool* tp, ThreadPoolJob* job, int* jobId) {
786 TRACE("Executing ThreadPoolAdd()")
787 int rc = EOUTOFMEM;
788 int tempId = -1;
789 long totalJobs;
790 ThreadPoolJob* temp = NULL;
791
792 if (!tp || !job)
793 return EINVAL;
794
795 ithread_mutex_lock(&tp->mutex);
796
797 totalJobs = tp->highJobQ.size + tp->lowJobQ.size + tp->medJobQ.size;
798 if (totalJobs >= tp->attr.maxJobsTotal) {
799 fprintf(stderr, "libupnp ThreadPoolAdd too many jobs: %ld\n",
800 totalJobs);
801 goto exit_function;
802 }
803 if (!jobId)
804 jobId = &tempId;
805 *jobId = INVALID_JOB_ID;
806 temp = CreateThreadPoolJob(job, tp->lastJobId, tp);
807 if (!temp)
808 goto exit_function;
809 switch (job->priority) {
810 case HIGH_PRIORITY:
811 if (ListAddTail(&tp->highJobQ, temp))
812 rc = 0;
813 break;
814 case MED_PRIORITY:
815 if (ListAddTail(&tp->medJobQ, temp))
816 rc = 0;
817 break;
818 default:
819 if (ListAddTail(&tp->lowJobQ, temp))
820 rc = 0;
821 }
822 /* AddWorker if appropriate */
823 AddWorker(tp);
824 /* Notify a waiting thread */
825 if (rc == 0)
826 ithread_cond_signal(&tp->condition);
827 else
828 FreeThreadPoolJob(tp, temp);
829 *jobId = tp->lastJobId++;
830
831exit_function:
832 ithread_mutex_unlock(&tp->mutex);
833
834 return rc;
835}

837int ThreadPoolRemove(ThreadPool* tp, int jobId, ThreadPoolJob* out) {
838 int ret = INVALID_JOB_ID;
839 ThreadPoolJob* temp = NULL;
840 ListNode* tempNode = NULL;
841 ThreadPoolJob dummy;
842
843 if (!tp)
844 return EINVAL;
845 if (!out)
846 out = &dummy;
847 dummy.jobId = jobId;
848
849 ithread_mutex_lock(&tp->mutex);
850
851 tempNode = ListFind(&tp->highJobQ, NULL, &dummy);
852 if (tempNode) {
853 temp = (ThreadPoolJob*)tempNode->item;
854 *out = *temp;
855 ListDelNode(&tp->highJobQ, tempNode, 0);
856 FreeThreadPoolJob(tp, temp);
857 ret = 0;
858 goto exit_function;
859 }
860
861 tempNode = ListFind(&tp->medJobQ, NULL, &dummy);
862 if (tempNode) {
863 temp = (ThreadPoolJob*)tempNode->item;
864 *out = *temp;
865 ListDelNode(&tp->medJobQ, tempNode, 0);
866 FreeThreadPoolJob(tp, temp);
867 ret = 0;
868 goto exit_function;
869 }
870 tempNode = ListFind(&tp->lowJobQ, NULL, &dummy);
871 if (tempNode) {
872 temp = (ThreadPoolJob*)tempNode->item;
873 *out = *temp;
874 ListDelNode(&tp->lowJobQ, tempNode, 0);
875 FreeThreadPoolJob(tp, temp);
876 ret = 0;
877 goto exit_function;
878 }
879 if (tp->persistentJob && tp->persistentJob->jobId == jobId) {
880 *out = *tp->persistentJob;
881 FreeThreadPoolJob(tp, tp->persistentJob);
882 tp->persistentJob = NULL;
883 ret = 0;
884 goto exit_function;
885 }
886
887exit_function:
888 ithread_mutex_unlock(&tp->mutex);
889
890 return ret;
891}

893int ThreadPoolGetAttr(ThreadPool* tp, ThreadPoolAttr* out) {
894 if (!tp || !out)
895 return EINVAL;
896 if (!tp->shutdown)
897 ithread_mutex_lock(&tp->mutex);

945int ThreadPoolShutdown(ThreadPool* tp) {
946 TRACE2("Executing ThreadPoolShutdown() for ThreadPool ", tp)
947 ListNode* head = NULL;
948 ThreadPoolJob* temp = NULL;
949
950 if (!tp)
951 return EINVAL;
952 ithread_mutex_lock(&tp->mutex);
953 /* clean up high priority jobs */
954 while (tp->highJobQ.size) {
955 head = ListHead(&tp->highJobQ);
956 if (head == NULL) {
957 ithread_mutex_unlock(&tp->mutex);
958 return EINVAL;
959 }
960 temp = (ThreadPoolJob*)head->item;
961 if (temp->free_func)
962 temp->free_func(temp->arg);
963 FreeThreadPoolJob(tp, temp);
964 ListDelNode(&tp->highJobQ, head, 0);
965 }
966 ListDestroy(&tp->highJobQ, 0);
967 /* clean up med priority jobs */
968 while (tp->medJobQ.size) {
969 head = ListHead(&tp->medJobQ);
970 if (head == NULL) {
971 ithread_mutex_unlock(&tp->mutex);
972 return EINVAL;
973 }
974 temp = (ThreadPoolJob*)head->item;
975 if (temp->free_func)
976 temp->free_func(temp->arg);
977 FreeThreadPoolJob(tp, temp);
978 ListDelNode(&tp->medJobQ, head, 0);
979 }
980 ListDestroy(&tp->medJobQ, 0);
981 /* clean up low priority jobs */
982 while (tp->lowJobQ.size) {
983 head = ListHead(&tp->lowJobQ);
984 if (head == NULL) {
985 ithread_mutex_unlock(&tp->mutex);
986 return EINVAL;
987 }
988 temp = (ThreadPoolJob*)head->item;
989 if (temp->free_func)
990 temp->free_func(temp->arg);
991 FreeThreadPoolJob(tp, temp);
992 ListDelNode(&tp->lowJobQ, head, 0);
993 }
994 ListDestroy(&tp->lowJobQ, 0);
995 /* clean up long term job */
996 if (tp->persistentJob) {
997 temp = tp->persistentJob;
998 if (temp->free_func)
999 temp->free_func(temp->arg);
1000 FreeThreadPoolJob(tp, temp);
1001 tp->persistentJob = NULL;
1002 }
1003 /* signal shutdown */
1004 tp->shutdown = 1;
1005 ithread_cond_broadcast(&tp->condition);
1006 /* wait for all threads to finish */
1007 while (tp->totalThreads > 0)
1008 ithread_cond_wait(&tp->start_and_shutdown, &tp->mutex);
1009 /* destroy condition */
1010 while (ithread_cond_destroy(&tp->condition) != 0) {
1011 }
1012 while (ithread_cond_destroy(&tp->start_and_shutdown) != 0) {
1013 }
1014 FreeListDestroy(&tp->jobFreeList);
1015
1016 ithread_mutex_unlock(&tp->mutex);
1017
1018 /* destroy mutex */
1019 while (ithread_mutex_destroy(&tp->mutex) != 0) {
1020 }
1021
1022 return 0;
1023}
