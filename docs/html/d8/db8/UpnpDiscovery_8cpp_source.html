95 ok = ok &&
96 UpnpDiscovery_set_ServiceVer(p, UpnpDiscovery_get_ServiceVer(q));
97 ok = ok && UpnpDiscovery_set_Location(p, UpnpDiscovery_get_Location(q));
98 ok = ok && UpnpDiscovery_set_Os(p, UpnpDiscovery_get_Os(q));
99 ok = ok && UpnpDiscovery_set_Date(p, UpnpDiscovery_get_Date(q));
100 ok = ok && UpnpDiscovery_set_Ext(p, UpnpDiscovery_get_Ext(q));
