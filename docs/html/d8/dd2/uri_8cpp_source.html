190int parse_hostport(
192 const char* in,
195 unsigned short int defaultPort,
197 hostport_type* out) {
198 char workbuf[256];
199 char* c;
200 struct sockaddr_in* sai4 = (struct sockaddr_in*)&out->IPaddress;
201 struct sockaddr_in6* sai6 = (struct sockaddr_in6*)&out->IPaddress;
202 char* srvname = NULL;
203 char* srvport = NULL;
204 char* last_dot = NULL;
205 unsigned short int port;
206 int af = AF_UNSPEC;
207 size_t hostport_size;
208 int has_port = 0;
209 int ret;
210
211 memset(out, 0, sizeof(hostport_type));
212 memset(workbuf, 0, sizeof(workbuf));
213 /* Work on a copy of the input string. */
214 strncpy(workbuf, in, sizeof(workbuf) - 1);
215 c = workbuf;
216 if (*c == '[') {
217 /* IPv6 addresses are enclosed in square brackets. */
218 srvname = ++c;
219 while (*c != '\0' && *c != ']')
220 c++;
221 if (*c == '\0')
222 /* did not find closing bracket. */
223 return UPNP_E_INVALID_URL;
224 /* NULL terminate the srvname and then increment c. */
225 *c++ = '\0'; /* overwrite the ']' */
226 if (*c == ':') {
227 has_port = 1;
228 c++;
229 }
230 af = AF_INET6;
231 } else {
232 /* IPv4 address -OR- host name. */
233 srvname = c;
234 while (*c != ':' && *c != '/' &&
235 (isalnum(*c) || *c == '.' || *c == '-')) {
236 if (*c == '.')
237 last_dot = c;
238 c++;
239 }
240 has_port = (*c == ':') ? 1 : 0;
241 /* NULL terminate the srvname */
242 *c = '\0';
243 if (has_port == 1)
244 c++;
245 if (last_dot != NULL && isdigit(*(last_dot + 1)))
246 /* Must be an IPv4 address. */
247 af = AF_INET;
248 else {
249 /* Must be a host name. */
250 struct addrinfo hints, *res, *res0;
251
252 memset(&hints, 0, sizeof(hints));
253 hints.ai_family = AF_UNSPEC;
254 hints.ai_socktype = SOCK_STREAM;
255
256 ret = umock::netdb_h.getaddrinfo(srvname, NULL, &hints, &res0);
257 if (ret == 0) {
258 for (res = res0; res; res = res->ai_next) {
259 switch (res->ai_family) {
260 case AF_INET:
261 case AF_INET6:
262 /* Found a valid IPv4 or IPv6
263 * address. */
264 memcpy(&out->IPaddress, res->ai_addr, res->ai_addrlen);
265 goto found;
266 }
267 }
268 found:
269 umock::netdb_h.freeaddrinfo(res0);
270 if (res == NULL)
271 /* Didn't find an AF_INET or AF_INET6
272 * address. */
273 return UPNP_E_INVALID_URL;
274 } else
275 /* getaddrinfo failed. */
276 return UPNP_E_INVALID_URL;
277 }
278 }
279 /* Check if a port is specified. */
280 if (has_port == 1) {
281 /* Port is specified. */
282 srvport = c;
283 while (*c != '\0' && isdigit(*c))
284 c++;
285 port = (unsigned short int)atoi(srvport);
286 if (port == 0)
287 /* Bad port number. */
288 return UPNP_E_INVALID_URL;
289 } else
290 /* Port was not specified, use default port. */
291 port = defaultPort;
292 /* The length of the host and port string can be calculated by */
293 /* subtracting pointers. */
294 hostport_size = (size_t)c - (size_t)workbuf;
295 /* Fill in the 'out' information. */
296 switch (af) {
297 case AF_INET:
298 sai4->sin_family = (sa_family_t)af;
299 sai4->sin_port = htons(port);
300 ret = inet_pton(AF_INET, srvname, &sai4->sin_addr);
301 break;
302 case AF_INET6:
303 sai6->sin6_family = (sa_family_t)af;
304 sai6->sin6_port = htons(port);
305 sai6->sin6_scope_id = gIF_INDEX;
306 ret = inet_pton(AF_INET6, srvname, &sai6->sin6_addr);
307 break;
308 default:
309 /* IP address was set by the hostname (getaddrinfo). */
310 /* Override port: */
311 if (out->IPaddress.ss_family == (sa_family_t)AF_INET)
312 sai4->sin_port = htons(port);
313 else
314 sai6->sin6_port = htons(port);
315 ret = 1;
316 }
317 /* Check if address was converted successfully. */
318 if (ret <= 0)
319 return UPNP_E_INVALID_URL;
320 out->text.size = hostport_size;
321 out->text.buff = in;
322
323 return (int)hostport_size;
324}

344size_t parse_scheme(
346 const char* in,
348 size_t max,
350 token* out) {
351 size_t i = (size_t)0;
352
353 out->size = (size_t)0;
354 out->buff = NULL;
355
356 if ((max == (size_t)0) || (!isalpha(in[0])))
357 return (size_t)0;
358
359 i++;
360 while ((i < max) && (in[i] != ':')) {
361 if (!(isalnum(in[i]) || (in[i] == '+') || (in[i] == '-') ||
362 (in[i] == '.')))
363 return (size_t)0;
364 i++;
365 }
366 if (i < max) {
367 out->size = i;
368 out->buff = &in[0];
369 return i;
370 }
371
372 return (size_t)0;
373}

391 int tempInt = 0;
392 char tempChar = 0;
393 size_t i = (size_t)0;
394 size_t j = (size_t)0;
395
396 if (in[index] == '%' && isxdigit(in[index + (size_t)1]) &&
397 isxdigit(in[index + (size_t)2])) {
398 /* Note the "%2x", makes sure that we convert a maximum of two
399 * characters. */
400#ifdef _WIN32
401 if (sscanf_s(&in[index + (size_t)1],
402#else
403 if (sscanf(&in[index + (size_t)1],
404#endif
405 "%2x", (unsigned int*)&tempInt) != 1) {
406 return 0;
407 }
408 tempChar = (char)tempInt;
409 for (i = index + (size_t)3, j = index; j < *max; i++, j++) {
410 in[j] = tempChar;
411 if (i < *max) {
412 tempChar = in[i];
413 } else {
414 tempChar = 0;
415 }
416 }
417 *max -= (size_t)2;
418 return 1;
419 } else {

532int remove_dots(char* buf, size_t size) {
533 char* in = buf;
534 char* out = buf;
535 char* max = buf + size;
536
537 while (!is_end_path(in[0])) {
538 assert(buf <= out);
539 assert(out <= in);
540 assert(in < max);
541
542 /* case 2.A: */
543 if (strncmp(in, "./", 2) == 0) {
544 in += 2;
545 } else if (strncmp(in, "../", 3) == 0) {
546 in += 3;
547 /* case 2.B: */
548 } else if (strncmp(in, "/./", 3) == 0) {
549 in += 2;
550 } else if (strncmp(in, "/.", 2) == 0 && is_end_path(in[2])) {
551 in += 1;
552 in[0] = '/';
553 /* case 2.C: */
554 } else if (strncmp(in, "/../", 4) == 0 ||
555 (strncmp(in, "/..", 3) == 0 && is_end_path(in[3]))) {
556 /* Make the next character in the input buffer a '/': */
557 if (is_end_path(in[3])) { /* terminating "/.." case */
558 in += 2;
559 in[0] = '/';
560 } else { /* "/../" prefix case */
561 in += 3;
562 }
563 /* Trim the last component from the output buffer, or
564 * empty it. */
565 while (buf < out)
566 if (*--out == '/')
567 break;
568#ifdef DEBUG
569 if (out < in)
570 out[0] = '\0';
571#endif
572 /* case 2.D: */
573 } else if (strncmp(in, ".", 1) == 0 && is_end_path(in[1])) {
574 in += 1;
575 } else if (strncmp(in, "..", 2) == 0 && is_end_path(in[2])) {
576 in += 2;
577 /* case 2.E */
578 } else {
579 /* move initial '/' character (if any) */
580 if (in[0] == '/')
581 *out++ = *in++;
582 /* move first segment up to, but not including, the next
583 * '/' character */
584 while (in < max && in[0] != '/' && !is_end_path(in[0]))
585 *out++ = *in++;
586#ifdef DEBUG
587 if (out < in)
588 out[0] = '\0';
589#endif
590 }
591 }
592 while (in < max)
593 *out++ = *in++;
594 if (out < max)
595 out[0] = '\0';
596 return UPNP_E_SUCCESS;
597}

599char* resolve_rel_url(char* base_url, char* rel_url) {
600 uri_type base;
601 uri_type rel;
602 int rv;
603 size_t len_rel;
604 size_t len_base;
605 size_t len;
606 char* out;
607 char* out_finger;
608 char* path;
609 size_t i;
610 size_t prefix;
611
612 if (!base_url) {
613 if (!rel_url)
614 return NULL;
615 return strdup(rel_url);
616 }
617
618 // BUG! Following segfaults if rel_url is NULL
619 len_rel = strlen(rel_url);
620 if (parse_uri(rel_url, len_rel, &rel) != HTTP_SUCCESS)
621 return NULL;
622 if (rel.type == (enum uriType)ABSOLUTE)
623 return strdup(rel_url);
624
625 len_base = strlen(base_url);
626 if ((parse_uri(base_url, len_base, &base) != HTTP_SUCCESS) ||
627 (base.type != (enum uriType)ABSOLUTE))
628 return NULL;
629 if (len_rel == (size_t)0)
630 return strdup(base_url);
631
632 len = len_base + len_rel + (size_t)2;
633 out = (char*)malloc(len);
634 if (out == NULL)
635 return NULL;
636 memset(out, 0, len);
637 out_finger = out;
638
639 /* scheme */
640 rv = snprintf(out_finger, len, "%.*s:", (int)base.scheme.size,
641 base.scheme.buff);
642 if (rv < 0 || rv >= (int)len)
643 goto error;
644 out_finger += rv;
645 len -= (size_t)rv;
646
647 /* authority */
648 if (rel.hostport.text.size > (size_t)0) {
649 rv = snprintf(out_finger, len, "%s", rel_url);
650 if (rv < 0 || rv >= (int)len)
651 goto error;
652 return out;
653 }
654 if (base.hostport.text.size > (size_t)0) {
655 rv = snprintf(out_finger, len, "//%.*s", (int)base.hostport.text.size,
656 base.hostport.text.buff);
657 if (rv < 0 || rv >= (int)len)
658 goto error;
659 out_finger += rv;
660 len -= (size_t)rv;
661 }
662
663 /* path */
664 path = out_finger;
665 if (rel.path_type == (pathType)ABS_PATH) {
666 rv = snprintf(out_finger, len, "%s", rel_url);
667 } else if (base.pathquery.size == (size_t)0) {
668 rv = snprintf(out_finger, len, "/%s", rel_url);
669 } else {
670 if (rel.pathquery.size == (size_t)0) {
671 rv = snprintf(out_finger, len, "%.*s", (int)base.pathquery.size,
672 base.pathquery.buff);
673 } else {
674 if (len < base.pathquery.size)
675 goto error;
676 i = 0;
677 prefix = 1;
678 while (i < base.pathquery.size) {
679 out_finger[i] = base.pathquery.buff[i];
680 switch (base.pathquery.buff[i++]) {
681 case '/':
682 prefix = i;
683 /* fall-through */
684 default:
685 continue;
686 case '?': /* query */
687 if (rel.pathquery.buff[0] == '?')
688 prefix = --i;
689 }
690 break;
691 }
692 out_finger += prefix;
693 len -= prefix;
694 rv = snprintf(out_finger, len, "%.*s", (int)rel.pathquery.size,
695 rel.pathquery.buff);
696 }
697 if (rv < 0 || rv >= (int)len)
698 goto error;
699 out_finger += rv;
700 len -= (size_t)rv;
701
702 /* fragment */
703 if (rel.fragment.size > (size_t)0)
704 rv = snprintf(out_finger, len, "#%.*s", (int)rel.fragment.size,
705 rel.fragment.buff);
706 else if (base.fragment.size > (size_t)0)
707 rv = snprintf(out_finger, len, "#%.*s", (int)base.fragment.size,
708 base.fragment.buff);
709 else
710 rv = 0;
711 }
712 if (rv < 0 || rv >= (int)len)
713 goto error;
714 out_finger += rv;
715 len -= (size_t)rv;
716
717 if (remove_dots(path, (size_t)(out_finger - path)) != UPNP_E_SUCCESS)
718 goto error;
719
720 return out;
721
722error:
723 free(out);
724 return NULL;
725}

727int parse_uri(const char* in, size_t max, uri_type* out) {
728 int begin_path = 0;
729 size_t begin_hostport = (size_t)0;
730 size_t begin_fragment = (size_t)0;
731 unsigned short int defaultPort = 80;
732
733 begin_hostport = parse_scheme(in, max, &out->scheme);
734 if (begin_hostport) {
735 out->type = (enum uriType)ABSOLUTE;
736 out->path_type = OPAQUE_PART;
737 begin_hostport++; // skip ':' scheme delimiter
738 } else {
739 out->type = (enum uriType)RELATIVE;
740 out->path_type = REL_PATH;
741 }
742 if (begin_hostport + (size_t)1 < max && in[begin_hostport] == '/' &&
743 in[begin_hostport + (size_t)1] == '/') {
744 begin_hostport += (size_t)2;
745 if (token_string_casecmp(&out->scheme, "https") == 0) {
746 defaultPort = 443;
747 }
748 begin_path =
749 parse_hostport(&in[begin_hostport], defaultPort, &out->hostport);
750 if (begin_path >= 0) {
751 begin_path += (int)begin_hostport;
752 } else
753 return begin_path; // error code from parse_hostport()
754 } else {
755 memset(&out->hostport, 0, sizeof(out->hostport));
756 begin_path = (int)begin_hostport;
757 }
758 begin_fragment =
759 parse_uric(&in[begin_path], max - (size_t)begin_path, &out->pathquery) +
760 (size_t)begin_path;
761 if (out->pathquery.size && out->pathquery.buff[0] == '/') {
762 out->path_type = ABS_PATH;
763 }
764 if (begin_fragment < max && in[begin_fragment] == '#') {
765 begin_fragment++;
766 parse_uric(&in[begin_fragment], max - begin_fragment, &out->fragment);
767 } else {
768 out->fragment.buff = NULL;
769 out->fragment.size = (size_t)0;
770 }
771
772 return HTTP_SUCCESS;
773}
